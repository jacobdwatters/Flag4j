package com.flag4j.concurrency.algorithms;

import com.flag4j.TypedMatrix;
import com.flag4j.concurrency.Configurations;

import java.lang.reflect.Constructor;
import java.lang.reflect.InvocationTargetException;
import java.util.ArrayList;
import java.util.List;

final class ThreadManager<M extends TypedMatrix, T extends Thread> {

    private List<Thread> threadList;
    private MatrixWorker worker;
    Constructor<?> clazz;

    protected ThreadManager(Class<?> worker) {
        try {
            this.clazz = worker.getConstructor(TypedMatrix.class, TypedMatrix.class,
                    Integer.class, Integer.class,
                    Integer.class, Integer.class);
        } catch (NoSuchMethodException e) {
            e.printStackTrace();
        }
    }


    protected List<Thread> allocateByRows(M src, M dest) {
        threadList = new ArrayList<>();
        int numThreads = Configurations.getNumThreads();
        int rowBlockSize, colBlockSize;
        int rowEnd, colEnd;

        if(src.numRows() < numThreads) {
            rowBlockSize = 1;
            colBlockSize = 1;
        } else {
            rowBlockSize = 8*src.numRows()/numThreads;
            colBlockSize = src.numCols()/8;
        }

        for(int i=0; i<src.numRows(); i += rowBlockSize) {
            for(int j=0; j<src.numCols(); j += colBlockSize) {
                rowEnd = Math.min(i+rowBlockSize, src.numRows());
                colEnd = Math.min(j+colBlockSize, src.numCols());

                try {
                    threadList.add((T) clazz.newInstance(dest, src, i, rowEnd, j, colEnd));
                } catch (InstantiationException e) {
                    e.printStackTrace();
                } catch (IllegalAccessException e) {
                    e.printStackTrace();
                } catch (InvocationTargetException e) {
                    e.printStackTrace();
                }

                threadList.get(threadList.size()-1).start(); // Start the thread
            }
        }

        return threadList;
    }

}
