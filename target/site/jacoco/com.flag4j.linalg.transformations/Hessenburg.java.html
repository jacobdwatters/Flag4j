package com.flag4j.linalg.transformations;


import com.flag4j.Matrix;
import com.flag4j.Vector;
import com.flag4j.util.ErrorMessages;
import com.flag4j.util.ParameterChecks;

/**
 * This class contains several methods for converting matrices to upper Hessenburg form using similarity transforms.
 */
public class Hessenburg {

    static final double EPS = 1.0E-10;

    private Hessenburg() {
        // Hide default constructor for utility class.
        throw new IllegalStateException(ErrorMessages.getUtilityClassErrMsg());
    }


    /**
     * Converts a real dense square matrix into upper Hessenburg form using similarity transforms.
     * @param src The matrix to compute the upper hessenburg form of.
     * @return An upper Hessenburg matrix which is similar to the {@code src} matrix.
     * @throws IllegalArgumentException If the {@code src} matrix is not square.
     */
    public static Matrix hessu(Matrix src) {
        ParameterChecks.assertSquare(src.shape);

        Matrix B = src.copy(); // Storage for upper Hessenburg matrix
        Matrix H; // For storing Householder reflector
        Vector col; // Normal vector for Householder reflector computation.
        Matrix Q = Matrix.I(B.numRows);

        for(int k=0; k<B.numRows-2; k++) {
            col = B.getColBelow(k+1, k).toVector();

            if(!col.isZeros()) { // If the column is zeros, no need to compute reflector. It is already in the correct form.
                H = Matrix.I(B.numRows);
                H.setSlice(Householder.getReflector(col), k+1, k+1);

                B = H.mult(B).multTranspose(H); // Apply Householder reflector to both sides of B.
                Q = Q.mult(H); // Apply Householder reflector to Q.
            }
        }

        System.out.println(Q.mult(B).mult(Q.T()));

        return B;
    }

    public static void main(String[] args) {
        double[][] a = {
                {1, 2, 3},
                {4, 5, 6},
                {7, 8, 9}};
        double[][] b = {
                {1, 0, 2, 3},
                {-1, 0, 5, 2},
                {2, -2, 0, 0},
                {2, -1, 2, 0}};
        Matrix A = new Matrix(a);
        Matrix B = new Matrix(b);


        Matrix H = hessu(B);
        System.out.println(H);
    }
}
