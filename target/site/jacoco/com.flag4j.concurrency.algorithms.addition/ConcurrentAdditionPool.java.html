package com.flag4j.concurrency.algorithms.addition;

import com.flag4j.Matrix;
import com.flag4j.concurrency.Configurations;
import com.flag4j.util.ErrorMessages;

import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.ThreadPoolExecutor;
import java.util.concurrent.TimeUnit;

public class ConcurrentAdditionPool {

    private static final int BASE_THRESHOLD = 9000;
    private static final int LARGER_THRESHOLD = 80000;
    private static final int SMALLER_THRESHOLD = 1000;

    private ConcurrentAdditionPool() {
        // Hide default constructor
        throw new IllegalStateException(ErrorMessages.utilityClassErrMsg());
    }


    /**
     * Computes the transpose of matrix A using multiple threads.
     * @param src1 First matrix to add.
     * @param src2 Second matrix to add.
     * @return The transpose of this matrix.
     */
    public static Matrix concurrentAdd(Matrix src1, Matrix src2) {
        int numThreads = Configurations.getNumThreads();
        int rowBlockSize, colBlockSize;
        int rowEnd, colEnd;

        Matrix dest = new Matrix(src1.numRows(), src1.numCols());
        ExecutorService executor = Executors.newFixedThreadPool(numThreads);
        ThreadPoolExecutor pool = (ThreadPoolExecutor) executor;

        if(src1.numRows() < numThreads && src1.numCols() < numThreads) {
            rowBlockSize = 1;
            colBlockSize = 1;

        } else if(src1.numRows() >= numThreads && src1.numCols() >= numThreads) {
            rowBlockSize = (int) Math.floor( ((double) src1.numRows()) / Math.sqrt(numThreads) );
            colBlockSize = (int) Math.floor( ((double) src1.numCols()) / Math.sqrt(numThreads) );

        } else if(src1.numRows() >= src1.numCols()) {
            rowBlockSize = Math.max(1, 4*src1.numRows()/numThreads);
            colBlockSize = Math.max(1, src1.numCols()/4);
        } else {
            colBlockSize = Math.max(1, 4*src1.numCols()/numThreads);
            rowBlockSize = Math.max(1, src1.numRows()/4);
        }

        for(int i=0; i<src1.numRows(); i += rowBlockSize) {
            for(int j=0; j<src1.numCols(); j += colBlockSize) {
                rowEnd = Math.min(i+rowBlockSize, src1.numRows());
                colEnd = Math.min(j+colBlockSize, src1.numCols());

                pool.execute(new MatrixAdditionPoolWorker(dest, src1, src2, i, rowEnd, j, colEnd));
            }
        }

        pool.shutdown();

        try {
            pool.awaitTermination(Integer.MAX_VALUE, TimeUnit.NANOSECONDS);
        } catch (InterruptedException e) {
            dest = null;
        }

        return dest;
    }


    public static boolean satisfiesConcurrent(int numRows, int numCols) {
        boolean result = false;

        if(Configurations.getNumThreads() > 1) {
            if(numRows >= BASE_THRESHOLD && numCols >= BASE_THRESHOLD) {
                result = true;
            } else if(numRows >= LARGER_THRESHOLD && numCols >= SMALLER_THRESHOLD) {
                result = true;
            } else if(numCols >= LARGER_THRESHOLD && numRows >= SMALLER_THRESHOLD) {
                result = true;
            }
        }

        return result;
    }

}
