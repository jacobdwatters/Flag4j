package com.flag4j.concurrency;


public class BuiltInOperations {

    public static final BinaryOperation<Double> addDouble = Double::sum;
    public static final BinaryOperation<Double> subDouble = (Double a, Double b) -> a-b;
    public static final BinaryOperation<Double> multDouble = (Double a, Double b) -> a*b;
    public static final BinaryOperation<Double> divDouble = (Double a, Double b) -> a/b;

    public static void main(String[] args) {
        Double x = 5.0;
        Double y = 3.3;

        Double z = subDouble.apply(x, y);

        System.out.println(z);
    }
}
