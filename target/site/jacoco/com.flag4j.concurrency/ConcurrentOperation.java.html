package com.flag4j.concurrency;

/**
 * A class for applying concurrent operations to matrices.
 * @param <T> Type of the entries of the matrix.
 * @param <M> Type of the matrix.
 */
public final class ConcurrentOperation {

    /**
     * Private constructor to hide from other files.
     */
    private ConcurrentOperation() {
        // Hides constructor
        throw new IllegalStateException(ErrorMessages.utilityClassErrMsg());
    }


    /**
     * Applies a closed element-wise binary operation to two matrices concurrently.
     * @param opp Operation to apply element-wise.
     * @param A First matrix in binary operation.
     * @param B Second matrix in binary operation.
     * @return The result of the element-wise binary operation
     */
    public static Matrix applyElementWiseConcurrent(BinaryOperation<Double> opp, Matrix A, Matrix B) {
        Matrix result = new Matrix(A.numRows(), A.numCols());
        List<Thread> threadList = allocateThreads(opp, result, A, B);

        for(Thread thread : threadList) { // Join the threads together
            try {
                thread.join();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }

        threadList.clear();

        return result;
    }

    /**
     * Allocate threads for binary operation on two real dense matrices.
     * @param opp Binary operation to apply.
     * @param dest Destination for result of binary matrix operation.
     * @param src1 First matrix in operation.
     * @param src2 Second matrix in operation.
     * @return A list of {@link MatrixBinaryOppWorker} threads which will concurrently transpose the matrix.
     */
    private static List<Thread> allocateThreads(BinaryOperation<Double> opp, Matrix dest, Matrix src1, Matrix src2) {
        List<Thread> threadList = new ArrayList<>();
        int numThreads = Configurations.getNumThreads();
        int rowBlockSize, colBlockSize;
        int rowEnd, colEnd;

        if(src1.numRows() < numThreads && src1.numCols() < numThreads) {
            rowBlockSize = 1;
            colBlockSize = 1;

        } else if(src1.numRows() >= numThreads && src1.numCols() >= numThreads) {
            rowBlockSize = (int) Math.floor( ((double) src1.numRows()) / Math.sqrt(numThreads) );
            colBlockSize = (int) Math.floor( ((double) src1.numCols()) / Math.sqrt(numThreads) );

        } else if(src1.numRows() >= src1.numCols()) {
            rowBlockSize = Math.max(1, 4*src1.numRows()/numThreads);
            colBlockSize = Math.max(1, src1.numCols()/4);
        } else {
            colBlockSize = Math.max(1, 4*src1.numCols()/numThreads);
            rowBlockSize = Math.max(1, src1.numRows()/4);
        }

        for(int i=0; i<src1.numRows(); i += rowBlockSize) {
            for(int j=0; j<src1.numCols(); j += colBlockSize) {
                rowEnd = Math.min(i+rowBlockSize, src1.numRows());
                colEnd = Math.min(j+colBlockSize, src1.numCols());

                threadList.add(new MatrixBinaryOppWorker(opp, dest, src1, src2, i, rowEnd, j, colEnd));
                threadList.get(threadList.size()-1).start(); // Start the thread
            }
        }

        return threadList;
    }
}
