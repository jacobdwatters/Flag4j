package com.flag4j.concurrency;

import com.flag4j.Matrix;

/**
 * A worker thread for computing an element-wise operation on a specified region of the matrix.
 */
class MatrixBinaryOppWorker extends Thread {

    BinaryOperation<Double> operation;
    Matrix dest;
    Matrix src1;
    Matrix src2;
    int rowStart;
    int rowEnd;
    int colStart;
    int colEnd;


    /**
     * Construct a worker thread for the specified binary operation.
     * @param operation Binary operation to be performed.
     * @param dest Destination matrix.
     * @param src1 First matrix in the operation.
     * @param src2 Second matrix in the operation.
     * @param rowStart Starting row of the matrix for the thread.
     * @param rowEnd Ending row of the matrix for the thread.
     * @param colStart Starting column of the matrix for the thread.
     * @param colEnd Ending column of the matrix for the thread.
     */
    protected MatrixBinaryOppWorker(BinaryOperation<Double> operation, Matrix dest, Matrix src1, Matrix src2,
                                    int rowStart, int rowEnd, int colStart, int colEnd) {

        this.operation = operation;
        this.dest = dest;
        this.src1 = src1;
        this.src2 = src2;
        this.rowStart = rowStart;
        this.rowEnd = rowEnd;
        this.colStart = colStart;
        this.colEnd = colEnd;
    }


    @Override
    public void run() {

        for(int i=rowStart; i<rowEnd; i++) {
            for(int j=colStart; j<colEnd; j++) {
                dest.entries[i][j] = operation.apply(src1.entries[i][j], src2.entries[i][j]);
            }
        }
    }
}
