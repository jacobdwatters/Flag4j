package com.flag4j;

import com.flag4j.operations.RealDenseOperations;
import com.flag4j.operations.concurrency.algorithms.addition.ConcurrentAddition;
import com.flag4j.util.RandomTensor;

import java.util.Random;

public class Test {

    public static void printArr(double[] arr) {
        for(double a : arr) {
            System.out.print(a + "  ");
        }
        System.out.println();
    }


    public static void arrEquals(double[] arr1, double[] arr2) {

        if(arr1.length != arr2.length) {
            throw new RuntimeException("Arrays not equal.");
        }

        for(int i=0; i<arr1.length; i++) {
            if(arr1[i] != arr2[i]) {
                throw new RuntimeException("Arrays not equal.");
            }
        }
    }


    public static void main(String args[]) {
        int rows = 12000;
        int cols = 15000;
        RandomTensor rand = new RandomTensor();
        Matrix A = rand.getRandomMatrix(rows, cols);
        Matrix B = rand.getRandomMatrix(rows, cols);

        long startTime, endTime;

        startTime = System.currentTimeMillis();
        Matrix sum1 = A.add(B);
        endTime = System.currentTimeMillis();
        System.out.println("Standard Add: " + (endTime - startTime) + " ms");

        startTime = System.currentTimeMillis();
        Matrix sum2 = ConcurrentAddition.add(A, B);
        endTime = System.currentTimeMillis();
        System.out.println("Concurrent Add: " + (endTime - startTime) + " ms");

        arrEquals(sum1.entries, sum2.entries);
    }
}
