package com.flag4j;


import com.flag4j.operations.dense.real.RealMatrixMultiplication;
import com.flag4j.util.RandomTensor;

import java.lang.reflect.Array;
import java.util.Arrays;

public class Testing {


    public static void printMat(Matrix mat) {

        for(int i=0; i<mat.numRows(); i++) {
            for(int j=0; j<mat.numCols(); j++) {
                System.out.print(mat.get(i, j));

                if(j!=mat.numCols()-1) {
                    System.out.print(", ");
                }
            }
            System.out.println();
        }
    }


    public static void main(String[] args) {
        long startTime, endTime;
        double[] re1, re2;
        RandomTensor rtg = new RandomTensor(42l);

        int numRows = 3000;
        int numCols = 3000;

        Matrix A = rtg.getRandomMatrix(numRows, numCols);

        startTime = System.nanoTime();
        re1 = RealMatrixMultiplication.standard(A.entries, A.shape, A.entries, A.shape);
        endTime = System.nanoTime();
        System.out.println(String.format("standard: %.2f ms", (endTime-startTime)*1.0e-6));

        startTime = System.nanoTime();
        RealMatrixMultiplication.standardSwapped(A.entries, A.shape, A.entries, A.shape);
        endTime = System.nanoTime();
        System.out.println(String.format("swapped: %.2f ms", (endTime-startTime)*1.0e-6));

        startTime = System.nanoTime();
        re2 = RealMatrixMultiplication.emjlReorder(A.entries, A.shape, A.entries, A.shape);
        endTime = System.nanoTime();
        System.out.println(String.format("emjlReorder: %.2f ms", (endTime-startTime)*1.0e-6));

        System.out.println(Arrays.equals(re1, re2));
    }
}
