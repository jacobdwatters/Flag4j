package com.flag4j;

import java.util.Random;


/**
 * Interface for specifying a lambda expression for an elementwiseOperation
 */
interface ElementWise {
    double execute(double a, double b);
}

class operations {
    static double addOpp(double a, double b) {
        return a+b;
    }
}

public class Main {
    static Random rng = new Random();

    public static Matrix applyElementWise(Matrix A, Matrix B, ElementWise opp) {
        double[][] sum = new double[A.m][A.n];
        for(int i=0; i<sum.length; i++) {
            for(int j=0; j<sum[0].length; j++) {
                sum[i][j] = opp.execute(A.entries[i][j], B.entries[i][j]);
            }
        }
        return new Matrix(sum);
    }

    public static Matrix add(Matrix A, Matrix B) {
        double[][] sum = new double[A.m][A.n];
        for(int i=0; i<sum.length; i++) {
            for(int j=0; j<sum[0].length; j++) {
                sum[i][j] = A.entries[i][j] + B.entries[i][j];
//                sum[i][j] = Double.sum(A.entries[i][j], B.entries[i][j]);
            }
        }
        return new Matrix(sum);
    }

    public static Matrix getRandom(int m, int n) {
        double[][] rand = new double[m][n];

        for(int i=0; i<m; i++) {
            for(int j=0; j<n; j++) {
                rand[i][j] = rng.nextDouble()*1000;
            }
        }

        return new Matrix(rand);
    }

    static void printMat(Matrix A) {
        for(int i=0; i<A.m; i++) {
            for(int j=0; j<A.n; j++) {
                System.out.print(A.entries[i][j]);

                if(j != A.n-1) {
                    System.out.print(", ");
                }
            }

            System.out.println();
        }
    }

    static void checkEq(Matrix A, Matrix B) {
        boolean eq = true;

        for(int i=0; i<A.m; i++) {
            for(int j=0; j<A.n; j++) {
                if(A.entries[i][j] != B.entries[i][j]) {
                    eq = false;
                }
            }
        }

        if(eq) {
            System.out.println("Results Match!");
        } else {
            System.err.println("ERROR: Results do not match.");
        }
    }

    public static void main(String[] args) {
        ElementWise add = (double a, double b) -> a+b;
        final int numRows = 14000;
        final int numCols = 14000;

//        Matrix A = new Matrix(new double[][]{{1, 2, 3}, {4, 5, 6}, {7, 8, 9}});
//        Matrix B = new Matrix(new double[][]{{9, 8, 7}, {6, 5, 4}, {3, 2, 1}});
        Matrix A = getRandom(numRows, numCols);
        Matrix B = getRandom(numRows, numCols);


        final long startTime1 = System.currentTimeMillis();
        Matrix C1 = applyElementWise(A, B, add);
        final long endTime1 = System.currentTimeMillis();
        final long eps1 = endTime1-startTime1;

        final long startTime2 = System.currentTimeMillis();
        Matrix C2 = add(A, B);
        final long endTime2 = System.currentTimeMillis();
        final long eps2 = endTime2-startTime2;

        checkEq(C1, C2);
        System.out.println(String.format("Lambda Time: %d ms", eps1));
        System.out.println(String.format("Standard Time: %d ms", eps2));


//        System.out.println("C1:\n");
//        printMat(C1);
//        System.out.println("\n");
//
//        System.out.println("C2:\n");
//        printMat(C2);
//        System.out.println("\n");
    }
}
