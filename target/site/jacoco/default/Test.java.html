import com.flag4j.Matrix;
import com.flag4j.concurrency.Configurations;
import com.flag4j.concurrency.algorithms.subtraction.ConcurrentSubtraction;
import com.flag4j.util.RandomTensor;

public class Test {


    public static Matrix standardAdd(Matrix A, Matrix B) {
        Matrix sum = new Matrix(A.numRows(), A.numCols());

        for(int i=0; i<sum.numRows(); i++) {
            for(int j=0; j< sum.numCols(); j++) {
                sum.entries[i][j] = A.entries[i][j] - B.entries[i][j];
            }
        }

        return sum;
    }



    public static Matrix blockedAdd(Matrix A, Matrix B) {
        Matrix sum = new Matrix(A.numRows(), A.numCols());

        int blockSize = Configurations.getBlockSize()/3;

        // Blocked matrix addition
        for(int i=0; i<sum.numRows(); i+=blockSize) {
            for(int j=0; j<sum.numCols(); j+=blockSize) {
                for(int k=i; k<i+blockSize && k<sum.numRows(); k++) {
                    for(int l=j; l<j+blockSize && l<sum.numCols(); l++) {
                        sum.entries[k][l] = A.entries[k][l] - B.entries[k][l];
                    }
                }
            }
        }

        return sum;
    }

    public static Matrix concurrentAdd(Matrix A, Matrix B) {
        Matrix sum = ConcurrentSubtraction.add(A, B);

        return sum;
    }


    public static void main(String[] args) {
        long startTime, endTime;

        int numRows = 10000;
        int numCols = 10000;

        int numRuns = 100;

        double[][] a = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
        double[][] b = {{9, 8, 7}, {6, 5, 4}, {3, 2, 1}};

        RandomTensor rand = new RandomTensor(42);
        Matrix A = rand.getRandomMatrix(numRows, numCols);
        Matrix B = rand.getRandomMatrix(numRows, numCols);
//        Matrix A = new Matrix(a);
//        Matrix B = new Matrix(b);

        startTime = System.currentTimeMillis();
        for(int i=0; i<numRuns; i++)
            concurrentAdd(A, B);
        endTime = System.currentTimeMillis();
        System.out.println("concurrentAdd: " + (endTime - startTime)/numRuns + " ms");

        startTime = System.currentTimeMillis();
        for(int i=0; i<numRuns; i++)
            standardAdd(A, B);
        endTime = System.currentTimeMillis();
        System.out.println("standardAdd: " + (endTime - startTime)/numRuns + " ms");

        startTime = System.currentTimeMillis();
        for(int i=0; i<numRuns; i++)
            blockedAdd(A, B);
        endTime = System.currentTimeMillis();
        System.out.println("blockedAdd: " + (endTime - startTime)/numRuns + " ms");

//        System.out.println(C1 + "\n\n");

    }
