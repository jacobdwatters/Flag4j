                new CNumber(1, 9),
                new CNumber(12, 40)};

        CNumber[] arr2 = Arrays.copyOfRange(arr, 0, 3);


        arr[0].re = 999.9999;

        System.out.println("arr: " + Arrays.toString(arr));
