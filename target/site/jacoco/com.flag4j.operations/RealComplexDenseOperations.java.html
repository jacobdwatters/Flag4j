package com.flag4j.operations;


import com.flag4j.complex_numbers.CNumber;
import com.flag4j.core.ComplexMatrixBase;
import com.flag4j.core.RealMatrixBase;
import com.flag4j.core.TensorBase;
import com.flag4j.util.ShapeChecks;

/**
 * This class provides methods for computing operations on one real and one complex tensor.
 */
public class RealComplexDenseOperations {


    /**
     * Computes the element-wise addition of two tensors.
     * @param A First Tensor of the addition.
     * @param B Second Tensor of the addition.
     * @return The element-wise addition of two tensors.
     * @throws IllegalArgumentException If A and B do not have the same shape.
     */
    public static CNumber[] add(RealMatrixBase A, ComplexMatrixBase B) {
        ShapeChecks.equalShapeCheck(A.shape, B.shape);

        CNumber[] sum = new CNumber[A.entries.length];

        for(int i=0; i<sum.length; i++) {
            sum[i] = new CNumber(
                    A.entries[i] + B.entries[i].re,
                    B.entries[i].im
            );
        }

        return sum;
    }


    /**
     * Computes the element-wise subtraction of two tensors.
     * @param A First Tensor of the addition.
     * @param B Second Tensor of the addition.
     * @return The element-wise subtraction of two tensors.
     * @throws IllegalArgumentException If A and B do not have the same shape.
     */
    public static CNumber[] sub(RealMatrixBase A, ComplexMatrixBase B) {
        ShapeChecks.equalShapeCheck(A.shape, B.shape);

        CNumber[] difference = new CNumber[A.entries.length];

        for(int i=0; i<difference.length; i++) {
            difference[i] = new CNumber(
                    A.entries[i]-B.entries[i].re,
                    -B.entries[i].im
            );
        }

        return difference;
    }
