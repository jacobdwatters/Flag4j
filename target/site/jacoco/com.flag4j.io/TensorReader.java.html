package com.flag4j.io;


import com.flag4j.Matrix;
import com.flag4j.util.RandomTensor;

import java.io.IOException;


/**
 * The TensorReader class provides several static methods for reading serialized
 * tensors, matrices, and vectors from a file.
 */
public class TensorReader {


    // TODO: Remove, this is temporary for testing.
    public static void main(String[] args) throws IOException, ClassNotFoundException {
        RandomTensor rng = new RandomTensor();
        Matrix A = rng.randnMatrix(1500, 698);

        TensorOutputStream out = new TensorOutputStream("testOutput.ser");
        out.write(A);
        out.close();

        TensorInputStream in = new TensorInputStream("testOutput.ser");
        Matrix B = in.readMatrix();

        System.out.printf("A %s:\n%s\n\n", A.shape, A);
        System.out.printf("B %s:\n%s\n\n", B.shape, B);

        long start = System.nanoTime();
        A.equals(B);
        long end = System.nanoTime();
        double total = ((double) end - (double) start)*1e-6;
        System.out.printf("Time: %f ms\n", total);

    }
}
