package com.flag4j.concurrency.algorithms.transpose;

import com.flag4j.Matrix;
import com.flag4j.concurrency.Configurations;

import java.util.ArrayList;
import java.util.List;

public final class Transpose {


    /**
     * Computes the transpose of matrix A using multiple threads.
     * @param A The matrix to compute the transpose of.
     * @return The transpose of this matrix.
     */
    public static Matrix concurrentT(Matrix A, boolean byRows) {
        Matrix T = new Matrix(A.numCols(), A.numRows());
        List<Thread> threadList;

        if(byRows) {
            threadList = allocateThreadsByRows(T, A);
        } else {
            threadList = allocateThreadsByRowsCols(T, A);
        }

        for(Thread thread : threadList) { // Join the threads together
            try {
                thread.join();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }

        threadList.clear();

        return T;
    }


    public static List<Thread> allocateThreadsByRows(Matrix dest, Matrix src) {
        List<Thread> threadList = new ArrayList<>();
        int numThreads = Configurations.getNumThreads();
        int rowBlockSize;
        int rowEnd;

        if(src.numRows() < numThreads) {
            rowBlockSize = 1;
        } else {
            rowBlockSize = src.numRows()/numThreads;
        }

        for(int i=0; i<src.numRows(); i += rowBlockSize) {
            rowEnd = Math.min(i+rowBlockSize, src.numRows());
            threadList.add(new MatrixTransposeWorker(dest, src, i, rowEnd, 0, src.numCols()));
            threadList.get(threadList.size()-1).start(); // Start the thread
        }

        return threadList;
    }


    public static List<Thread> allocateThreadsByRowsCols(Matrix dest, Matrix src) {
        List<Thread> threadList = new ArrayList<>();
        int numThreads = Configurations.getNumThreads();
        int rowBlockSize, colBlockSize;
        int rowEnd, colEnd;

        if(src.numRows() < numThreads && src.numCols() < numThreads) {
            rowBlockSize = 1;
            colBlockSize = 1;

        } else {
            rowBlockSize = 4*src.numRows()/numThreads;
            colBlockSize = Math.max(1, src.numCols()/4);
        }

        for(int i=0; i<src.numRows(); i += rowBlockSize) {
            for(int j=0; j<src.numCols(); j += colBlockSize) {
                rowEnd = Math.min(i+rowBlockSize, src.numRows());
                colEnd = Math.min(j+colBlockSize, src.numCols());

                threadList.add(new MatrixTransposeWorker(dest, src, i, rowEnd, j, colEnd));
                threadList.get(threadList.size()-1).start(); // Start the thread
            }
        }

        return threadList;
    }
}
