30 import com.flag4j.complex_numbers.CNumber;

33 /**
34  * This solver solves linear systems of equations where the coefficient matrix in a lower triangular complex dense matrix
35  * and the constant vector is a complex dense vector.
36  */
37 public class ComplexForwardSolver implements LinearSolver<CMatrix, CVector, CVector> {
38 
39     /**
40      * Performs forward substitution for a unit lower triangular matrix {@code L} and a vector {@code b}.
41      * That is, solves the linear system {@code L*x=b} where {@code L} is a lower triangular matrix.
42      * @param L Unit lower triangular coefficient matrix. If {@code L} is not unit lower triangular, it will be treated
43      *          as such.
44      * @param b Constant vector.
45      * @return The result of solving the linear system {@code L*x=b} where {@code L} is a lower triangular matrix.
46      * @throws IllegalArgumentException If {@code L} is not lower triangular.
47      */
48     @Override
49     public CVector solve(CMatrix L, CVector b) {
50         CNumber sum;
51         int lIndexStart;
52         CVector x = new CVector(L.numRows);
53 
54         x.entries[0] = b.entries[0].copy();
55 
56         for(int i=1; i<L.numRows; i++) {
57             sum = new CNumber();
58             lIndexStart = i*L.numCols;
59 
60             for(int j=i-1; j>-1; j--) {
61                 sum.addEq(L.entries[lIndexStart + j].mult(x.entries[j]));
62             }
63 
64             x.entries[i] = b.entries[i].sub(sum);
65         }
66 
67         return x;
68     }
69 }
