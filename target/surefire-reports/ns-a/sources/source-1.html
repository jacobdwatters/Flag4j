31 /**
32  * This solver solves linear systems of equations where the coefficient matrix in an upper triangular complex dense matrix
33  * and the constant vector is a complex dense vector.
34  */
35 public class ComplexBackSolver implements LinearSolver<CMatrix, CVector, CVector> {
36 
37     // TODO: If a diagonal entry is zero, back-solve fails, add error should be thrown (system is singular).
38 
39     /**
40      * Solves the linear system of equations given by {@code U*x=b} where the coefficient matrix {@code U}
41      * is an upper triangular matrix.
42      *
43      * @param U Upper triangular coefficient matrix in the linear system. If {@code U} is not upper triangular, it
44      *          will be treated as such.
45      * @param b Vector of constants in the linear system.
46      * @return The solution to {@code x} in the linear system {@code A*x=b}.
47      */
48     @Override
49     public CVector solve(CMatrix U, CVector b) {
50         CNumber sum;
51         int uIndex;
52         int n = b.size;
53         CVector x = new CVector(U.numRows);
54 
55         x.entries[n-1] = b.entries[n-1].div(U.entries[n*n-1]);
56 
57         for(int i=n-2; i>-1; i--) {
58             sum = new CNumber();
59             uIndex = i*U.numCols;
60 
61             for(int j=i+1; j<n; j++) {
62                 sum.addEq(U.entries[uIndex + j].mult(x.entries[j]));
63             }
64 
65             x.entries[i] = (b.entries[i].sub(sum)).div(U.entries[i*(n+1)]);
66         }
67 
68         return x;
69     }
70 }
