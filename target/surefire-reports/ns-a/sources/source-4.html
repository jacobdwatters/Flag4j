32 /**
33  * <p>Solves a well determined system of equations {@code Ax=b} in an exact sense by using a {@code LU} decomposition.</p>
34  * <p>If the system is not well determined, i.e. {@code A} is square and full rank, then use a
35  * {@link LstsqSolver least-squares solver}.</p>
36  */
37 public abstract class LUSolver<T extends MatrixMixin<T, ?, ?, ?, ?, ?, ?>, U, V>
38         implements LinearSolver<T, U, V> {
39 
40     /**
41      * Decomposer to compute {@code LU} decomposition.
42      */
43     protected final LUDecomposition<T> lu;
44     /**
45      * Unit lower triangular matrix in {@code} LU decomposition.
46      */
47     protected T L;
48     /**
49      * Upper triangular matrix in {@code} LU decomposition.
50      */
51     protected T U;
52     /**
53      * Row permutation matrix for {@code LU} decomposition.
54      */
55     protected Matrix P;
56 
57     /**
58      * Constructs an exact LU solver with a specified {@code LU} decomposer.
59      * @param lu {@code LU} decomposer to employ in solving the linear system.
60      * @throws IllegalArgumentException If the {@code LU} decomposer does not use partial pivoting.
61      */
62     protected LUSolver(LUDecomposition<T> lu) {
63         if(lu.pivotFlag!=LUDecomposition.Pivoting.PARTIAL) {
64             throw new IllegalArgumentException("LU solver must use partial pivoting but got " +
65                     lu.pivotFlag.name() + ".");
66         }
67 
68         this.lu = lu;
69     }
70 
71 
72     /**
73      * Decomposes A using an {@link LUDecomposition LU decomposition}.
74      * @param A Matrix to decompose.
75      */
76     protected void decompose(T A) {
77         lu.decompose(A);
78         L = lu.getL();
79         U = lu.getU();
80         P = lu.getP();
81     }
82 }
