34 /**
35  * This class solves a linear system of equations {@code Ax=b} in a least-squares sense. That is,
36  * minimizes {@code ||Ax-b||<sub>2</sub>} which is equivalent to solving the normal equations
37  * {@code A<sup>T</sup>Ax=A<sup>T</sup>b}.
38  * This is done using a {@link QRDecomposition}.
39  */
40 public class RealLstsqSolver extends LstsqSolver<Matrix, Vector, Vector> {
41 
42     /**
43      * Backwards solver for solving the system of equations formed from the {@code QR} decomposition,
44      * {@code Rx=Q<sup>T</sup>b} which is an equivalent system to {@code A<sup>T</sup>Ax=A<sup>T</sup>b}.
45      */
46     private final RealBackSolver backSolver;
47 
48     /**
49      * Constructs a least-squares solver to solve a system {@code Ax=b} in a least square sense. That is,
50      * minimizes {@code ||Ax-b||<sub>2</sub>} which is equivalent to solving the normal equations
51      * {@code A<sup>T</sup>Ax=A<sup>T</sup>b}.
52      */
53     protected RealLstsqSolver() {
54         super(new RealQRDecomposition());
55         backSolver = new RealBackSolver();
56     }
57 
58 
59     /**
60      * Solves the linear system given by {@code Ax=b} in the least-squares sense.
61      *
62      * @param A Coefficient matrix in the linear system.
63      * @param b Vector of constants in the linear system.
64      * @return The least squares solution to {@code x} in the linear system {@code Ax=b}.
65      */
66     @Override
67     public Vector solve(Matrix A, Vector b) {
68         decompose(A); // Compute the QR decomposition of A.
69         return backSolver.solve(R, Q.T().mult(b).toVector());
70     }
71 }
