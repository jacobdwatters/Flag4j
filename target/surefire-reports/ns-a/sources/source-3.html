56     /**
57      * Solves the linear system of equations given by {@code A*x=b} for {@code x}.
58      *
59      * @param A Coefficient matrix in the linear system.
60      * @param b Vector of constants in the linear system.
61      * @return The solution to {@code x} in the linear system {@code A*x=b}.
62      */
63     @Override
64     public CVector solve(CMatrix A, CVector b) {
65         ParameterChecks.assertSquare(A.shape); // Ensure A is square.
66         ParameterChecks.assertEquals(A.numCols, b.size); // b must have the same number of entries as columns in A.
67         decompose(A); // Compute the decomposition of the coefficient matrix.
68 
69         double det = detLU(L, U).magAsDouble();
70 
71         if(det <= RANK_THRESHOLD || Double.isNaN(det)) {
72             throw new IllegalArgumentException("Matrix expected to have full rank but did not.");
73         }
74 
75         CVector y = forwardSolver.solve(L, P.mult(b).toVector());
76         return backSolver.solve(U, y);
77     }
