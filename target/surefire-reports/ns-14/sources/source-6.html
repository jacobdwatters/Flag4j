377      * Generates a symmetric matrix filled with pseudorandom values uniformly distributed in {@code [0, 1)}.
378      * @param size Number of rows and columns in the resulting matrix (the result will be a square matrix).
379      * @return A symmetric matrix filled with pseudorandom values uniformly distributed in {@code [0, 1)}.
380      */
381     public Matrix getRandomSymmetricMatrix(int size) {
382         Matrix randMat = new Matrix(size);
383 
384         for(int i=0; i<size; i++) {
385             for(int j=0; j<i; j++) {
386                 randMat.entries[i*size+j] = complexRng.rng.nextDouble();
387                 randMat.entries[j*size+i] = randMat.entries[i*size+j];
388             }
389 
390             randMat.entries[i*(size+1)] = complexRng.rng.nextDouble(); // Diagonal entry
391         }
392 
393         return randMat;
394     }
