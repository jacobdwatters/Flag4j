47     /**
48      * Multiplies a real dense matrix to the transpose of a complex sparse matrix.
49      * @param dSrc Entries of dense matrix.
50      * @param dShape Shape of dense matrix.
51      * @param spSrc Non-zero entries of sparse matrix.
52      * @param rowIndices Row indices of non-zero entries in the sparse matrix.
53      * @param colIndices Column indices of non-zero entries in the sparse matrix.
54      * @param spShape Shape of the sparse matrix.
55      * @return The entries of the matrix resulting from multiplying the first matrix by the transpose of the second matrix.
56      */
57     public static CNumber[] multTranspose(double[] dSrc, Shape dShape,
58                                           CNumber[] spSrc, int[] rowIndices, int[] colIndices, Shape spShape) {
59         int rows1 = dShape.dims[Axis2D.row()];
60         int rows2 = spShape.dims[Axis2D.row()];
61         int cols2 = spShape.dims[Axis2D.col()];
62 
63         CNumber[] dest = new CNumber[rows1*rows2]; // Since second matrix is transposed, its columns will become rows.
64         ArrayUtils.fillZeros(dest);
65 
66         int row, col;
67         int destStart, dSrcStart;
68 
69         for(int i=0; i<rows1; i++) {
70             destStart = i*rows2;
71             dSrcStart = i*cols2;
72 
73             // Loop over non-zero entries of sparse matrix.
74             for(int j=0; j<spSrc.length; j++) {
75                 row = colIndices[j];
76                 col = rowIndices[j];
77 
78                 dest[destStart + col].addEq(spSrc[j].mult(dSrc[dSrcStart + row]));
79             }
80         }
81 
82         return dest;
83     }
