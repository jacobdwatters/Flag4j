49      * Computes the matrix multiplication between a real dense matrix and a complex sparse matrix using a standard algorithm.
50      * @param src1 Entries of the dense matrix.
51      * @param shape1 Shape of the dense matrix.
52      * @param src2 Non-zero entries of the sparse matrix.
53      * @param rowIndices Row indices for non-zero entries of the sparse matrix.
54      * @param colIndices Column indices for non-zero entries of the sparse matrix.
55      * @param shape2 Shape of the sparse matrix.
56      * @return The result of the matrix multiplication.
57      */
58     public static CNumber[] standard(double[] src1, Shape shape1, CNumber[] src2,
59                                     int[] rowIndices, int[] colIndices, Shape shape2) {
60         int rows1 = shape1.dims[Axis2D.row()];
61         int cols1 = shape1.dims[Axis2D.col()];
62         int cols2 = shape2.dims[Axis2D.col()];
63 
64         CNumber[] dest = new CNumber[rows1*cols2];
65         ArrayUtils.fill(dest, CNumber.ZERO);
66 
67         int row, col;
68 
69         for(int i=0; i<rows1; i++) {
70             // Loop over non-zero entries of sparse matrix.
71             for(int j=0; j<src2.length; j++) {
72                 row = rowIndices[j];
73                 col = colIndices[j];
74 
75                 dest[i*cols2 + col].addEq(src2[j].mult(src1[i*cols1 + row]));
76             }
77         }
78 
79         return dest;
80     }

379      * @param shape1 Shape of the dense matrix.
380      * @param src2 Non-zero entries of the sparse vector.
381      * @param indices Indices of non-zero entries in sparse vector.
382      * @return Entries of the dense matrix resulting from the matrix vector multiplication.
383      */
384     public static CNumber[] blockedVector(double[] src1, Shape shape1, CNumber[] src2, int[] indices) {
385         int rows1 = shape1.dims[Axis2D.row()];
386         int cols1 = shape1.dims[Axis2D.col()];
387         int rows2 = src2.length;
388 
389         int bsize = Configurations.getBlockSize(); // Get the block size to use.
390 
391         CNumber[] dest = new CNumber[rows1];
392         ArrayUtils.fill(dest, CNumber.ZERO);
393 
394         int k;
395 
396         // Blocked matrix-vector multiply
397         for(int ii=0; ii<rows1; ii += bsize) {
398             for(int jj=0; jj<rows2; jj += bsize) {
399                 // Multiply the current blocks
400                 for(int i=ii; i<ii+bsize && i<rows1; i++) {
401                     for(int j=jj; j<jj+bsize && j<rows2; j++) {
402                         k = indices[j];
403                         dest[i].addEq(src2[j].mult(src1[i*cols1 + k]));
404                     }
405                 }
406             }
407         }
408 
409         return dest;
410     }
