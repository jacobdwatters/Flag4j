33 /**
34  * <p>
35  *     The base class for all vectors. A vector is equivalent to a {@link TensorBase tensor} of rank 1. Vectors do not
36  *     have an orientation (that is, row/column vector). However, several methods assume or allow the user to specify the
37  *     orientation of the vector.
38  * </p>
39  * @param <T> Type of the storage data structure for the vector.
40  *           This common use case will be an array or list-like data structure.
41  */
42 public abstract class VectorBase<T extends Serializable> extends TensorBase<T> {
43 
44     /**
45      * Size of the vector. This includes zero-entries if the vector is sparse.
46      */
47     public final int size;
48 
49     /**
50      * Constructs a basic vector with the specified number of entries.
51      *
52      * @param size        Number of entries in this vector.
53      * @param entries     The non-zero entries of this sparse tensor.
54      */
55     public VectorBase(int size, T entries) {
56         super(new Shape(size), entries);
57         this.size = size;
58     }
59 
60 
61     /**
62      * Constructs a basic vector with the specified number of entries.
63      *
64      * @param shape        Number of entries in this vector.
65      * @param entries     The non-zero entries of this sparse tensor.
66      * @throws IllegalArgumentException If the rank of the shape is not 1.
67      */
68     public VectorBase(Shape shape, T entries) {
69         super(shape, entries);
70         ParameterChecks.assertRank(1, shape); // Ensure the shape is of rank 1.
71         this.size = shape.get(0);
72     }

75     /**
76      * Gets the size of this vector.
77      *

79      */

81         return size;
82     }
