170     /**
171      * Constructs a sparse tensor whose shape and values are given by another sparse tensor. This effectively copies
172      * the tensor.
173      * @param A Tensor to copy.
174      */
175     public SparseTensor(SparseTensor A) {
176         super(A.shape.copy(), A.nonZeroEntries(), A.entries.clone(), new int[A.indices.length][A.indices[0].length]);
177         for(int i=0; i<indices.length; i++) {
178             super.indices[i] = A.indices[i].clone();
179         }
180     }
