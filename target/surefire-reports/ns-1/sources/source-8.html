605         ParameterChecks.assertEqualShape(this.shape, b.shape);
606 
607         CNumber[] entries = new CNumber[this.entries.length + b.entries.length];
608         int[][] indices = new int[2][this.indices.length + b.indices.length]; // Row and column indices.
609 
610         // Copy values from this vector.
611         ArrayUtils.arraycopy(this.entries, 0, entries, 0, this.entries.length);
612         // Copy values from vector b.
613         ArrayUtils.arraycopy(b.entries, 0, entries, this.entries.length, b.entries.length);
614 
615         // Set row indices to 1 for b values (this vectors row indices are 0 which was implicitly set already).
616         Arrays.fill(indices[0], this.indices.length, entries.length, 1);
617 
618         // Copy indices from this vector to the column indices.
619         System.arraycopy(this.indices, 0, indices[1], 0, this.entries.length);
620         // Copy indices from b vector to the column indices.
621         System.arraycopy(b.indices, 0, indices[1], this.entries.length, b.entries.length);
622 
623         return new SparseCMatrix(new Shape(2, this.size), entries, indices[0], indices[1]);
624     }
