112      * Constructs strides for each dimension of this shape as if for a newly constructed tensor.
113      * i.e. Strides will be a decreasing sequence with the last stride being 1.
114      * @return The strides for all dimensions of a newly constructed tensor with this shape.
115      */
116     public int[] createNewStrides() {
117         int[] strides = new int[dims.length];
118 
119         if(strides.length>0) {
120             // Stride along last axis is always one for new strides.
121             strides[strides.length-1] = 1;
122 
123             for(int i=strides.length-2; i>=0; i--) {
124                 strides[i] = dims[i+1]*strides[i+1];
125             }
126         }
127 
128         return strides;
129     }

132     /**
133      * Computes the index of the 1D data array for a dense tensor from tensor indices with this shape.
134      * @param indices Indices of tensor with this shape.
135      * @return The index of the element at the specified indices in the 1D data array of a dense tensor.
136      * @throws IllegalArgumentException If the number of indices does not match the rank of this shape.
137      * @throws IndexOutOfBoundsException If any index does not fit within a tensor with this shape.
138      */
139     public int entriesIndex(int... indices) {
140         if(indices.length != dims.length) {
141             throw new IllegalArgumentException(ErrorMessages.getIndicesRankErr(indices.length, dims.length));
142         }
143         if(indices.length>0 && indices[indices.length-1] >= dims[dims.length-1]) {
144             throw new IndexOutOfBoundsException("Index " + indices[indices.length-1] + " out of bounds for axis " +
145                     (indices.length-1) + " of tensor with shape " + this);
146         }
147 
148         int index = 0;
149 
150         for(int i=0; i<indices.length-1; i++) {
151             if(indices[i] < 0 || indices[i] >= dims[i]) {
152                 throw new IndexOutOfBoundsException("Index " + indices[i] + " out of bounds for axis " + i +
153                         " of tensor with shape " + this);
154             }
155 
156             index += indices[i]*strides[i];
157         }
158 
159         return index + indices[indices.length-1];
160     }

169     public int[] getIndices(int index) {

171         indices[indices.length-1] = index % dims[dims.length-1];
172         int upStream = index;
173
