611      * <p>
612      * Stacks two vectors along specified axis.
613      * </p>
614      *
615      * <p>
616      * Stacking two vectors of length {@code n} along axis 0 stacks the vectors
617      * as if they were row vectors resulting in a {@code 2-by-n} matrix.
618      * </p>
619      *
620      * <p>
621      * Stacking two vectors of length {@code n} along axis 1 stacks the vectors
622      * as if they were column vectors resulting in a {@code n-by-2} matrix.
623      * </p>
624      *
625      * @param b    Vector to stack with this vector.
626      * @param axis Axis along which to stack vectors. If {@code axis=0}, then vectors are stacked as if they are row
627      *             vectors. If {@code axis=1}, then vectors are stacked as if they are column vectors.
628      * @return The result of stacking this vector and the vector {@code b}.
629      * @throws IllegalArgumentException If the number of entries in this vector is different from the number of
630      *                                  entries in the vector {@code b}.
631      * @throws IllegalArgumentException If axis is not either 0 or 1.
632      */
633     @Override
634     public CMatrix stack(Vector b, int axis) {
635         ParameterChecks.assertAxis2D(axis);
636         CMatrix stacked;
637 
638         if(axis==0) {
639             stacked = stack(b);
640         } else {
641             ParameterChecks.assertArrayLengthsEq(this.size, b.size);
642             CNumber[] stackedEntries = new CNumber[2*this.size];
643 
644             int count = 0;
645
