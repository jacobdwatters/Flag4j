158      * Checks if two this vector is numerically element-wise equal to another object. Objects which can be equal are, {@link Vector},
159      * {@link CVector}, {@link SparseVector}, {@link SparseCVector}.
160      *
161      * @param b Object to compare to this vector.
162      * @return True if this vector and object {@code b} are equivalent element-wise. Otherwise, returns false.
163      */
164     @Override
165     public boolean equals(Object b) {
166         boolean equal = false;
167 
168         if(b instanceof Vector) {
169             Vector vec = (Vector) b;
170             equal = ArrayUtils.equals(vec.entries, this.entries);
171         } else if(b instanceof CVector) {
172             CVector vec = (CVector) b;
173             equal = Arrays.equals(this.entries, vec.entries);
174         } else if(b instanceof SparseVector) {
175             SparseVector vec = (SparseVector) b;
176             equal = RealComplexDenseSparseEquals.vectorEquals(this.entries, vec.entries, vec.indices, vec.size);
177         } else if(b instanceof SparseCVector) {
178             SparseCVector vec = (SparseCVector) b;
179             equal = ComplexDenseSparseEquals.vectorEquals(this.entries, vec.entries, vec.indices, vec.size);
180         }
181 
182         return equal;
183     }

794      *          - If p is {@link Double#POSITIVE_INFINITY}, then this method computes the maximum/infinite norm.<br>

796      * @return The p-norm of this tensor.
797      */
798     @Override
799     public double norm(double p) {
800         if(Double.isInfinite(p)) {

802                 return maxAbs().re; // Maximum / infinite norm.
803             } else {
804                 return minAbs().re; // Minimum norm.
