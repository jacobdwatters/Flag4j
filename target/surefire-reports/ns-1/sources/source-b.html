169      * Sets an index of a copy of this vector to a specified value.
170      *
171      * Creates a copy of this vector and sets an index to the specified value. Note, unlike the dense version of this
172      * method, this <b>does not</b> affect this vector.
173      *
174      * @param value   Value to set.
175      * @param indices The index of for which to set the value.
176      * @return A copy of this vector with the specified index set to the specified value.
177      * @throws IllegalArgumentException If the number of indices provided is greater than 1.
178      * @throws IllegalArgumentException If the index is negative or larger than the total vector size.
179      */
180     @Override
181     public SparseVector set(double value, int... indices) {
182         ParameterChecks.assertEquals(indices.length, 1);
183         ParameterChecks.assertInRange(indices[0], 0, size, "index");
184 
185         int idx = Arrays.binarySearch(this.indices, indices[0]);
186         double[] destEntries;
187         int[] destIndices;
188 
189         if(idx >= 0) {
190             // Then the index was found in the sparse vector.
191             destIndices = this.indices.clone();
192             destEntries = entries.clone();
193             destEntries[idx] = value;
194 
195         } else{
196             // Then the index was not found int the sparse vector.
197             destIndices = new int[this.indices.length+1];
198             destEntries = new double[entries.length+1];
199             idx = -(idx+1);
200 
201             System.arraycopy(this.indices, 0, destIndices, 0, idx);
202             destIndices[idx] = indices[0];
203             System.arraycopy(this.indices, idx, destIndices, idx+1, this.indices.length-idx);
204 
205             System.arraycopy(entries, 0, destEntries, 0, idx);
206             destEntries[idx] = value;
207             System.arraycopy(entries, idx, destEntries, idx+1, entries.length-idx);
208         }
209 
210         return new SparseVector(size, destEntries, destIndices);
211     }

1107     /**
1108      * Checks if a vector is parallel to this vector. This sparse vectors indices are assumed to be sorted lexicographically.
1109      * If this is not the case call {@link #sparseSort() this.sparseSort()} before calling this method.
1110      *
1111      * @param b Vector to compare to this vector.
1112      * @return True if the vector {@code b} is parallel to this vector and the same size. Otherwise, returns false.
1113      */
1114     @Override
1115     public boolean isParallel(Vector b) {
1116         final double tol = 1.0e-12; // Tolerance to accommodate floating point arithmetic error in scaling.
1117         boolean result;
1118 
1119         if(this.size!=b.size) {
1120             result = false;
1121         } else if(this.size<=1) {
1122             result = true;
1123         } else if(this.isZeros() || b.isZeros()) {
1124             result = true; // Any vector is parallel to zero vector.
1125         } else {
1126             result = true;
1127             int sparseIndex = 0;
1128             double scale = 0;
1129 
1130             // Find first non-zero entry in b and compute the scaling factor (we know there is at least one from else-if).
1131             for(int i=0; i<b.size; i++) {
1132                 if(b.entries[i]!=0) {
1133                     scale = this.entries[i]/b.entries[this.indices[i]];
1134                     break;
1135                 }
1136             }
1137 
1138             for(int i=0; i<b.size; i++) {
1139                 if(sparseIndex >= this.entries.length || i!=this.indices[sparseIndex]) {
1140                     // Then this index is not in the sparse vector.
1141                     if(b.entries[i] != 0) {
1142                         result = false;
1143                         break;
1144                     }
1145 
1146                 } else {
1147                     // Ensure the scaled entry is approximately equal to the entry in this vector.
1148                     if(Math.abs(b.entries[i]*scale - this.entries[sparseIndex]) > tol) {
1149                         result = false;
1150                         break;
1151                     }
1152 
1153                     sparseIndex++;
1154                 }
1155             }
1156         }
1157 
1158         return result;
1159     }
