207      * Checks if an object is equal to this tensor object. Valid object types are: {@link Tensor}, {@link CTensor},
208      * {@link SparseTensor}, and {@link SparseCTensor}. These tensors are equal to this tensor if all entries are
209      * numerically equal to the corresponding element of this tensor. If the tensor is complex, then the imaginary
210      * component must be zero to be equal.
211      * @param object Object to check equality with this tensor.
212      * @return True if the two tensors are numerically equivalent and false otherwise.
213      */
214     @Override
215     public boolean equals(Object object) {
216         boolean equal;
217 
218         if(object instanceof Tensor) {
219             Tensor tensor = (Tensor) object;
220             equal = RealDenseEquals.tensorEquals(this, tensor);
221         } else if(object instanceof CTensor) {
222             CTensor tensor = (CTensor) object;
223             equal = RealComplexDenseEquals.tensorEquals(this, tensor);
224 
225         } else if(object instanceof SparseTensor) {
226             SparseTensor tensor = (SparseTensor) object;
227             equal = RealDenseSparseEquals.tensorEquals(this, tensor);
228 
229         } else if(object instanceof SparseCTensor) {
230             SparseCTensor tensor = (SparseCTensor) object;
231             equal = RealComplexDenseSparseEquals.tensorEquals(this, tensor);
232 
233         } else {
234             equal = false;
235         }
236 
237         return equal;
238     }
