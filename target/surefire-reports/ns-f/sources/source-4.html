47      * Computes the vector inner product between a real dense vector and a real sparse vector.
48      * @param src1 Entries of the dense vector.
49      * @param src2 Non-zero entries of the sparse vector.
50      * @param indices Indices of non-zero entries in the sparse vector.
51      * @param sparseSize The size of the sparse vector (including zero entries).
52      * @return The inner product of the two vectors.
53      * @throws IllegalArgumentException If the number of entries in the two vectors is not equivalent.
54      */
55     public static double innerProduct(double[] src1, double[] src2, int[] indices, int sparseSize) {
56         ParameterChecks.assertArrayLengthsEq(src1.length, sparseSize);
57         double innerProd = 0;
58         int index;
59 
60         for(int i=0; i<src2.length; i++) {
61             index = indices[i];
62 
63             innerProd += src1[index]*src2[i];
64         }
65 
66         return innerProd;
67     }

71      * Computes the vector outer product between a real dense vector and a real sparse vector.
72      * @param src1 Entries of the dense vector.
73      * @param src2 Non-zero entries of the sparse vector.
74      * @param indices Indices of non-zero entries of sparse vector.
75      * @param sparseSize Full size of the sparse vector including zero entries.
76      * @return The matrix resulting from the vector outer product.
77      */
78     public static double[] outerProduct(double[] src1, double[] src2, int[] indices, int sparseSize) {
79         double[] dest = new double[src1.length*sparseSize];
80         int index;
81

87             }
