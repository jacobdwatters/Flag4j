33 /**
34  * This class contains low-level implementations of complex element-wise tensor multiplication.
35  */
36 public class ComplexDenseElemMult {

38     /**
39      * Minimum number of entries in each tensor to apply concurrent algorithm.
40      */
41     private static final int CONCURRENT_THRESHOLD = 50625;
42 
43 
44     private ComplexDenseElemMult() {

46         throw new IllegalStateException(ErrorMessages.getUtilityClassErrMsg());
47     }

50     /**
51      * Computes the element-wise multiplication of two tensors. Also called the Hadamard product.
52      * @param src1 First tensor in element-wise multiplication.
53      * @param shape1 Shape of the first tensor.
54      * @param src2 Second tensor in element-wise multiplication.
55      * @param shape2 Shape of the second tensor.
56      * @return The element-wise multiplication of the two tensors.
57      * @throws IllegalArgumentException If the tensors do not have the same shape.
58      */
59     public static CNumber[] elemMult(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
60         ParameterChecks.assertEqualShape(shape1, shape2);
61         CNumber[] product = new CNumber[src1.length];
62 
63         for(int i=0; i<product.length; i++) {
64             product[i] = src1[i].mult(src2[i]);
65         }
66 
67         return product;
68     }

71     /**
72      * Computes the element-wise multiplication of two tensors using a concurrent algorithm. Also called the Hadamard product.
73      * @param src1 First tensor in element-wise multiplication.
74      * @param shape1 Shape of the first tensor.
75      * @param src2 Second tensor in element-wise multiplication.
76      * @param shape2 Shape of the second tensor.
77      * @return The element-wise multiplication of the two tensors.
78      * @throws IllegalArgumentException If the tensors do not have the same shape.
79      */
80     public static CNumber[] elemMultConcurrent(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
81         ParameterChecks.assertEqualShape(shape1, shape2);
82         CNumber[] product = new CNumber[src1.length];
83 
84         ThreadManager.concurrentLoop(0, product.length,
85                 (i)->product[i] = src1[i].mult(src2[i])
86         );
87 
88         return product;
89     }

92     /**
93      * Dynamically chooses and applies element-wise multiplication algorithm to use based on the number of entries in the tensors.
94      * @param src1 Entries of first tensor.
95      * @param shape1 Shape of first tensor.
96      * @param src2 Entries of second tensor.
97      * @param shape2 Shape of second tensor.
98      * @return The element-wise multiplication of the two tensors.
99      */
100     public static CNumber[] dispatch(CNumber[] src1, Shape shape1,CNumber[] src2, Shape shape2) {
101         if(src1.length < CONCURRENT_THRESHOLD) {
102             return elemMult(src1, shape1, src2, shape2);
103         } else {
104             return elemMultConcurrent(src1, shape1, src2, shape2);
105         }
106     }
