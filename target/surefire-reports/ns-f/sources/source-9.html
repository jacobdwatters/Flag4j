43      * Checks if this tensor only contains ones.
44      * @param src Elements of the tensor.
45      * @return True if this tensor only contains ones. Otherwise, returns false.
46      */
47     public static boolean isOnes(CNumber[] src) {
48         boolean allZeros = true;
49

52                 allZeros = false;

57         return allZeros;

61     /**
62      * Checks if a complex dense matrix is hermation. That is, if the and equal to its conjugate transpose.
63      * @param src Entries of the matrix.
64      * @param shape Shape of the matrix.
65      * @return True if this matrix is hermation. Otherwise, returns false.
66      */
67     public static boolean isHermation(CNumber[] src, Shape shape) {
68         if(shape.dims[0]!=shape.dims[1]) {
69             return false;
70         }
71 
72         int count1, count2, stop;
73 
74         for(int i=0; i<shape.dims[0]; i++) {
75             count1 = i*shape.dims[1];
76             count2 = i;
77             stop = count1 + i;
78 
79             while(count1 < stop) {
80                 if(src[count1++].equals(src[count2].conj())) {
81                     return false;
82                 }
83 
84                 count2+=shape.dims[1];
85             }
86         }
87 
88         return true;
89     }
