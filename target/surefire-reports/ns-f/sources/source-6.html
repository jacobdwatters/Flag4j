50     /**
51      * Multiplies a matrix to the transpose of a second matrix. This method may be significantly faster than computing the
52      * transpose and multiplication in two steps.
53      * @param src1 Entries of the first matrix.
54      * @param shape1 Shape of the first matrix.
55      * @param src2 Entries of the second matrix.
56      * @param shape2 Shape of the second matrix.
57      * @return The result of multiplying the first matrix with the transpose of the second matrix.
58      */
59     public static CNumber[] multTranspose(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
60         int rows1 = shape1.dims[Axis2D.row()];
61         int rows2 = shape2.dims[Axis2D.row()];
62         int cols2 = shape2.dims[Axis2D.col()];
63 
64         CNumber[] dest = new CNumber[rows1*rows2]; // Since second matrix is transposed, its columns will become rows.
65         ArrayUtils.fillZeros(dest);
66 
67         int src1Index, src2Index, destIndex, src1IndexStart, destIndexStart, end;
68 
69         for(int i=0; i<rows1; i++) {
70             src1IndexStart = i*cols2;
71             destIndexStart = i*rows2;
72             end = src1IndexStart + cols2;
73 
74             for(int j=0; j<rows2; j++) {
75                 src1Index = src1IndexStart;
76                 src2Index = j*cols2;
77                 destIndex = destIndexStart + j;
78 
79                 while(src1Index<end) {
80                     dest[destIndex].addEq(src1[src1Index++].mult(src2[src2Index++]));
81                 }
82             }
83         }
84 
85         return dest;
86     }

98     public static CNumber[] multTransposeBlocked(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
99         int rows1 = shape1.dims[Axis2D.row()];
100         int rows2 = shape2.dims[Axis2D.row()];
101         int cols2 = shape2.dims[Axis2D.col()];
102 
103         CNumber[] dest = new CNumber[rows1*rows2];
104         ArrayUtils.fillZeros(dest);
105 
106         int blockSize = Configurations.getBlockSize();
107         int iBound, jBound, kBound;
108         int src1Start, destStart, end;
109         int destIndex, src1Index, src2Index;
110 
111         for(int ii=0; ii<rows1; ii+=blockSize) {
112             iBound = Math.min(ii + blockSize, rows1);
113 
114             for(int jj = 0; jj<rows2; jj+=blockSize) {
115                 jBound = Math.min(jj + blockSize, rows2);
116 
117                 for(int kk = 0; kk<cols2; kk+=blockSize) {
118                     kBound = Math.min(kk + blockSize, cols2);
119 
120                     // Multiply the blocks
121                     for(int i=ii; i<iBound; i++) {
122                         destStart = i*rows2;
123                         src1Start = i*cols2 + kk;
124                         end = src1Start + kBound - kk;
125 
126                         for(int j=jj; j<jBound; j++) {
127                             destIndex = destStart + j;
128                             src1Index = src1Start;
129                             src2Index = j*cols2 + kk;

132                                 dest[destIndex].addEq(src1[src1Index++].mult(src2[src2Index++]));
