32  * <p>An instance of this class allows for the computation of a Cholesky decomposition for a
33  * real dense {@link Matrix matrix}.</p>
34  *
35  * <p>Given a symmetric positive-definite matrix {@code A}, the Cholesky decomposition will decompose it into
36  * {@code A=LL<sup>T</sup>} where {@code L} is a lower triangular matrix and {@code L<sup>T</sup>} is the
37  * transpose of {@code L}.</p>
38  */

44      * @return The transpose of the {@code L} matrix from the Cholesky decomposition {@code A=LL<sup>T</sup>}.
45      */

53      * Decompose a matrix into {@code A=LL<sup>T</sup>} where {@code L} is a lower triangular matrix and {@code L<sup>T</sup>} is the
54      * transpose of {@code L}.
55      *
56      * @param src The source matrix to decompose. Must be symmetric positive-definite.
57      * @return A reference to this decomposer.
58      * @throws IllegalArgumentException If {@code src} is not symmetric positive-definite.
59      */
60     @Override
61     public RealCholeskyDecomposition decompose(Matrix src) {
62         // TODO: Check that matrix is positive definite.
63         ParameterChecks.assertSquare(src.shape);
64 
65         L = new Matrix(src.numRows);
66         double sum;
67 
68         int lIndex1;
69         int lIndex2;
70         int lIndex3;
71 
72         for(int i=0; i<src.numCols; i++) {
73             lIndex1 = i*L.numCols;
74 
75             for(int j=0; j<=i; j++) {
76                 sum = 0;
77                 lIndex2 = j*L.numCols;
78                 lIndex3 = lIndex1 + j;
79 
80                 for(int k=0; k<j; k++) {
81                     sum += L.entries[lIndex1 + k]*L.entries[lIndex2 + k];
82                 }
83 
84                 if(i==j) {
85                     L.entries[lIndex3] = Math.sqrt(src.entries[lIndex3]-sum);
86                 } else {
87                     if(L.entries[j*(L.numCols + 1)] != 0) {
88                         L.entries[lIndex3] = (src.entries[lIndex3]-sum)/L.entries[lIndex2 + j];
89                     }
90                 }
91             }
92         }
93 
94         return this;
95     }
