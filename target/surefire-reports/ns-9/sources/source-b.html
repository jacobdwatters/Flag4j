32 /**
33  * <p>Computes the Hessenburg decomposition of a real dense square matrix. That is, for a square matrix
34  * {@code A}, computes the decomposition {@code A=QHQ<sup>T</sup>} where {@code Q} is an orthogonal matrix and
35  * {@code H} is a matrix in upper Hessenburg form which is similar to {@code A} (i.e. has the same eigenvalues/vectors).</p>
36  *
37  * <p>A matrix {@code H} is in upper Hessenburg form if it is nearly upper triangular. Specifically, if {@code H} has
38  * all zeros below the first sub-diagonal.</p>
39  *
40  * <p>For example, the following matrix is in upper Hessenburg form where each {@code x} may hold a different value:
41  * <pre>
42  *     [[ x x x x x ]
43  *      [ x x x x x ]
44  *      [ 0 x x x x ]
45  *      [ 0 0 x x x ]
46  *      [ 0 0 0 x x ]]</pre>
47  * </p>
48  */
49 public final class RealHessenburgDecomposition extends HessenburgDecomposition<Matrix> {

55      * create a Hessenburg decomposer which <b>does not</b> compute {@code Q}, see {@link #RealHessenburgDecomposition(boolean)}.
56      */

74     /**
75      * Hessenburg decomposition of a real dense square matrix. That is, for a square matrix
76      * {@code A}, computes the decomposition {@code A=QHQ<sup>T</sup>} where {@code Q} is an orthogonal matrix and
77      * {@code H} is a matrix in upper Hessenburg form which is similar to {@code A} (i.e. has the same eigenvalues/vectors).</p>
78      *
79      * @param src The source matrix to decompose.
80      * @return A reference to this decomposer.
81      * @throws IllegalArgumentException If the {@code src} matrix is not square.
82      */
83     @Override
84     public Decomposition<Matrix> decompose(Matrix src) {
85         ParameterChecks.assertSquare(src.shape);
86 
87         // TODO: Add a decomposition for symmetric matrices.
88         generalDecomposition(src);
89 
90         return this;
91     }

94     /**
95      * Computes the Hessenburg decomposition of a general real matrix.
96      * @param src Matrix to compute the Hessenburg decomposition of.
97      */
98     private void generalDecomposition(Matrix src) {
99         H = src.copy(); // Storage for upper Hessenburg matrix
100         Matrix ref; // For storing Householder reflector
101         Vector col; // Normal vector for Householder reflector computation.
102 
103         if(computeQ) {
104             Q = Matrix.I(this.H.numRows); // Storage for orthogonal matrix in the decomposition.
105         } else {
106             Q = null;
107         }
108 
109         for(int k = 0; k<this.H.numRows-2; k++) {
110             col = this.H.getColBelow(k+1, k).toVector();
111 
112             if(!col.isZeros()) { // If the column is zeros, no need to compute reflector. It is already in the correct form.
113                 ref = Matrix.I(this.H.numRows);
114                 ref.setSlice(Householder.getReflector(col), k+1, k+1);
115 
116                 H = ref.mult(H).multTranspose(ref); // Apply Householder reflector to both sides of H.
117 
118                 if(computeQ) {
119                     Q = Q.mult(ref); // Apply Householder reflector to Q.
120                 }
121             }
122         }
123     }
