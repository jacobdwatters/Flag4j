30 /**
31  * <p>This abstract class specifies methods for computing the Hessenburg decomposition of a square matrix. That is, for a square matrix
32  * {@code A}, computes the decomposition {@code A=QHQ<sup>H</sup>} where {@code Q} is a unitary matrix and
33  * {@code H} is a matrix in upper Hessenburg form which is similar to {@code A} (i.e. has the same eigenvalues/vectors).</p>
34  *
35  * <p>A matrix {@code H} is in upper Hessenburg form if it is nearly upper triangular. Specifically, if {@code H} has
36  * all zeros below the first sub-diagonal.</p>
37  *
38  * <p>For example, the following matrix is in upper Hessenburg form where each {@code x} may hold a different value:
39  * <pre>
40  *     [[ x x x x x ]
41  *      [ x x x x x ]
42  *      [ 0 x x x x ]
43  *      [ 0 0 x x x ]
44  *      [ 0 0 0 x x ]]</pre>
45  * </p>
46  */
47 public abstract class HessenburgDecomposition<T extends MatrixMixin<T, ?, ?, ?, ?, ?, ?>> implements Decomposition<T> {
48 
49     /**
50      * A flag for determining if {@code Q} should be computed in the Hessenburg decomposition.
51      */
52     protected boolean computeQ;
53 
54     /**
55      * Storage for the {@code Q} matrix in the Hessenburg decomposition corresponding to {@code A=QHQ<sup>H</sup>}.
56      */
57     protected T Q;
58     /**
59      * Storage for the {@code H} matrix in the Hessenburg decomposition corresponding to {@code A=QHQ<sup>H</sup>}.
60      */
61     protected T H;
62 
63 
64     /**
65      * Constructs a Hessenburg decomposer which specifies if the unitary matrix {@code Q} in the decomposition should
66      * be computed.
67      * @param computeQ Flag for determining if the {@code Q} in the decomposition should be computed.<br>
68      *                 - If true, the unitary {@code Q} matrix will be computed.<br>
69      *                 - If false, the unitary {@code Q} matrix will not be computed which may give a performance
70      *                 increase if it is not needed.
71      */
72     protected HessenburgDecomposition(boolean computeQ) {

74     }
