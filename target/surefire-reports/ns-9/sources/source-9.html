32 /**
33  * <p>This abstract class specifies methods for computing the LU decomposition of a matrix.</p>
34  * <p>The {@code LU} decomposition, decomposes a matrix {@code A} into a unit lower triangular matrix {@code L}
35  * and an upper triangular matrix {@code U} such that {@code A=LU}.</p>
36  * <p>If partial pivoting is used, the decomposition will also yield a permutation matrix {@code P} such that
37  * {@code PA=LU}.</p>
38  * <p>If full pivoting is used, the decomposition will yield an additional permutation matrix {@code Q} such that
39  *  {@code PAQ=LU}.</p>
40  */
41 public abstract class LUDecomposition<T extends MatrixMixin<T, ?, ?, ?, ?, ?, ?>> implements Decomposition<T> {

44      * Flag indicating what pivoting to use.
45      */
46     public final Pivoting pivotFlag;
47     /**
48      * for determining if pivot value is to be considered zero in LU decomposition with no pivoting.
49      */
50     final double DEFAULT_ZERO_PIVOT_TOLERANCE = 0.5e-14;
51     /**
52      * Tolerance for determining if pivot value is to be considered zero in LU decomposition with no pivoting.
53      */
54     final double zeroPivotTol;
55     /**
56      * Storage for L and U matrices. Stored in a single matrix
57      */
58     protected T LU;
59     /**
60      * Permutation matrix to store row swaps if partial pivoting is used.
61      */
62     protected Matrix P;
63     /**
64      * Permutation matrix to store column swaps if full pivoting is used.
65      */
66     protected Matrix Q;

69     /**
70      * Constructs a LU decomposer to decompose the specified matrix.
71      * @param pivoting Pivoting to use. If pivoting is 2, full pivoting will be used. If pivoting is 1, partial pivoting
72      *                 will be used. If pivoting is any other value, no pivoting will be used.
73      */
74     public LUDecomposition(int pivoting) {

76         zeroPivotTol = DEFAULT_ZERO_PIVOT_TOLERANCE;
77     }

82      * @param pivoting Pivoting to use. If pivoting is 2, full pivoting will be used. If pivoting is 1, partial pivoting
83      *                 will be used. If pivoting is any other value, no pivoting will be used.
84      * @param zeroPivotTol Tolerance for considering a pivot to be zero. If a pivot is less than the tolerance in absolute value,
85      *                     then it will be considered zero.
86      */
87     public LUDecomposition(int pivoting, double zeroPivotTol) {
88         pivotFlag = Pivoting.get(pivoting);
89         this.zeroPivotTol = zeroPivotTol;
90     }

93     /**
94      * Applies {@code LU} decomposition to the source matrix using the pivoting specified in the constructor.
95      *
96      * @param src The source matrix to decompose. Not modified.
97      * @return A reference to this decomposer.
98      */
99     @Override
100     public LUDecomposition<T> decompose(T src) {
101         initLU(src);
102 
103         if(pivotFlag==Pivoting.NONE) {
104             noPivot(); // Compute with no pivoting.
105         } else if(pivotFlag==Pivoting.PARTIAL) {
106             partialPivot();
107         } else {
108             fullPivot();
109         }
110 
111         return this;
112     }
