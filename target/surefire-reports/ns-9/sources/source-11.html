36 /**
37  * <p>This abstract class specifies methods for computing the Schur decomposition of a square matrix.
38  * That is, decompose a square matrix {@code A} into {@code A=UTU<sup>H</sup>} where {@code U} is a unitary
39  * matrix whose columns are the eigenvectors of {@code A} and {@code T} is an upper triangular matrix in
40  * Schur form whose diagonal entries are the eigenvalues of {@code A}, corresponding to the columns of {@code U},
41  * repeated per their multiplicity.</p>
42  *
43  * <p>Note, even if a matrix has only real entries, both {@code U} and {@code T} may contain complex values.</p>
44  *
45  * @param <T> The type of matrix to compute the Schur decomposition of.
46  */

139     /**
140      * Computes the Schur decomposition for a complex dense matrix using a double shifted implicit QR algorithm
141      * (also known as Francis's Algorithm of degree two).
142      * @param H The matrix to compute the Schur decomposition of. Assumed to be in upper Hessenburg form.
143      */
144     // TODO: This is a very rudimentary implementation of Francis's double shift implicit QR algorithm.
145     //  Several improvements must be made. See Fundamentals of matrix computations Vol. 3 for more information.
146     //  this may also only work for real symmetric matrices...
147     //  Currently, only computes the real schur form.
148     protected void doubleShiftImplicitQR(CMatrix H) {
149         if(useDefaultMaxIterations) {
150             maxIterations = Math.max(10*H.numRows, MIN_DEFAULT_ITERATIONS);
151         }
152 
153         ComplexHessenburgDecomposition hess = new ComplexHessenburgDecomposition(computeU); // Would need to compute Q to compute eigenvectors as well.
154 
155         T = H;
156         if(computeU) U = CMatrix.I(T.numRows);
157 
158         for(int i=0; i<maxIterations; i++) {
159             // Compute shifts as eigenvalues of lower right 2x2 block.
160             // TODO: Ensure this is numerically stable. Also, add random shifts as noted in Fundamentals of Matrix Computations Vol. 3.
161             CVector rho = Eigen.get2x2LowerLeftBlockEigenValues(T);
162 
163             CNumber[] pEntries = {
164                     T.entries[0].sub(rho.entries[0]).mult(T.entries[0].sub(rho.entries[1])).add(T.entries[1].mult(T.entries[T.numCols])),
165                     T.entries[T.numCols].mult( T.entries[0].add(T.entries[T.numCols + 1]).sub(rho.entries[0]).sub(rho.entries[1]) ),
166                     T.entries[2*T.numCols+1].mult(T.entries[T.numCols])
167             };
168             Vector p = new CVector(pEntries).toReal(); // Should be real anyway
169             Matrix Q = Householder.getReflector(p);
170 
171             // Apply the Householder reflector to T.
172             T.setSlice(Q.T().mult(T.getSlice(0, Q.numRows, 0, T.numCols)), 0, 0);
173             T.setSlice(T.getSlice(0, T.numRows, 0, Q.numCols).mult(Q), 0, 0);
174 
175             // TODO: A crude bulge chase using Hessenberg decomposition. Since we know A is nearly in Hessenberg already, this
176             //  will perform a lot of superfluous computations. Use Householder transformations instead.
177             T = hess.decompose(T).getH();
178 
179             if(computeU) {
180                 // Apply the Householder reflectors to U.
181                 U.setSlice(U.getSlice(0, U.numRows, 0, Q.numCols).mult(Q), 0, 0);
182                 U = U.mult(hess.getQ());
183             }
184 
185             if(T.roundToZero().isTriU()) {
186                 break; // We have converged (approximately) to an upper triangular matrix.
187             }
188         }
189     }

192     /**
193      * Computes the Schur decomposition for a complex matrix using a shifted QR algorithm where the QR decomposition
194      * @param H - The matrix to compute the Schur decomposition of. Assumed to be in upper Hessenburg form.
195      */
196     protected void shiftedExplicitQR(CMatrix H) {
197         if(useDefaultMaxIterations) {
198             maxIterations = Math.max(10*H.numRows, MIN_DEFAULT_ITERATIONS);
199         }
200 
201         QRDecomposition<CMatrix> qr = new ComplexQRDecomposition();
202 
203         // Initialize matrices.
204         T = H;
205         if(computeU) U = CMatrix.I(T.numRows);
206 
207         for(int i=0; i<maxIterations; i++) {
208             CNumber lam = getWilkinsonShift(T);
209             CMatrix mu = Matrix.I(T.numRows).mult(lam);
210 
211             // Compute the QR decomposition with a shift.
212             qr.decompose(T.sub(mu));
213 
214             // Update T and undo the shift.
215             T = qr.R.mult(qr.Q).add(mu);
216 
217             if(computeU) U = U.mult(qr.Q);
218 
219             if(T.roundToZero().isTriU()) {
220                 break; // We have converged (approximately) to an upper triangular matrix. No need to continue.
221             }
222         }
223     }

226     /**
227      * Computes the Wilkinson shift for a complex matrix. That is, the eigenvalue of the lower left 2x2 sub matrix
228      * which is closest in magnitude to the lower left entry of the matrix.
229      * @param src Matrix to compute the Wilkinson shift for.
230      * @return The Wilkinson shift for the {@code src} matrix.
231      */
232     private CNumber getWilkinsonShift(CMatrix src) {
233         // Compute eigenvalues of lower 2x2 block.
234         CVector lambdas = Eigen.get2x2LowerLeftBlockEigenValues(src);
235         CNumber v = src.entries[src.entries.length-1];
236 
237         if(v.sub(lambdas.entries[0]).magAsDouble() < v.sub(lambdas.entries[1]).magAsDouble()) {
238             return lambdas.entries[0];
239         } else {
240             return lambdas.entries[1];
241         }
242     }
