35  * <p>Instances of this class compute the {@code QR} decomposition of a complex dense matrix.</p>
36  * <p>The {@code QR} decomposition, decomposes a matrix {@code A} into an unitary matrix {@code Q}
37  * and an upper triangular matrix {@code R} such that {@code A=QR}.</p>
38  */

54      */
55     public ComplexQRDecomposition(boolean fullQR) {

60     /**
61      * Computes the reduced QR decomposition on the src matrix.
62      * @param src The source matrix to decompose.
63      */
64     @Override
65     protected void reduced(CMatrix src) {
66         full(src); // First compute the full decomposition
67 
68         int k = Math.min(src.numRows, src.numCols);
69 
70         // Now reduce the decomposition
71         Q = Q.getSlice(0, src.numRows, 0, k);
72         R = R.getSlice(0, k, 0, src.numCols);
73     }
74 
75 
76     /**
77      * Computes the full QR decomposition on the src matrix.
78      * @param src The source matrix to decompose.
79      */
80     @Override
81     protected void full(CMatrix src) {
82         R = new CMatrix(src); // Initialize R to the values in src.
83         int m = R.numRows, n = R.numCols;
84         int stop = Math.min(n, m-1);
85 
86         double eps = 1.0e-12;
87 
88         CMatrix H;
89         CVector col;
90 
91         // Initialize Q to the identity matrix.
92         Q = CMatrix.I(R.numRows);
93 
94         for(int i=0; i<stop; i++) {
95             col = R.getColBelow(i, i).toVector();
96 
97             // If the column has zeros below the diagonal it is in the correct form. No need to compute reflector.
98             if(col.maxAbs() > eps) {
99                 H = CMatrix.I(m);
100                 H.setSlice(Householder.getReflector(col), i, i);
101 
102                 Q = Q.mult(H); // Apply Householder reflector to Q
103                 R = H.mult(R); // Apply Householder reflector to R
104             }
105         }
106     }
