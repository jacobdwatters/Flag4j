29 /**
30  * <p>This abstract class specifies methods for computing the {@code QR} decomposition of a matrix.</p>
31  * <p>The {@code QR} decomposition, decomposes a matrix {@code A} into a unitary matrix {@code Q}
32  * and an upper triangular matrix {@code R} such that {@code A=QR}.</p>
33  */
34 public abstract class QRDecomposition<T extends MatrixMixin<T, ?, ?, ?, ?, ?, ?>> implements Decomposition<T> {
35 
36     /**
37      * Storage matrix for {@code Q}.
38      */
39     protected T Q;
40     /**
41      * Storage matrix for {@code R}.
42      */
43     protected T R;
44     /**
45      * Flag for determining if the full {@code QR} decomposition should be used. If true, the full {@code QR} decomposition will be computed,
46      * if false, the reduced QR decomposition will be computed.
47      */
48     protected final boolean fullQR;
49 
50     /**
51      * Constructs {@code QR} decomposer which computes the full {@code QR} decomposition.
52      */
53     public QRDecomposition() {
54         fullQR = true;
55     }
56 
57     /**
58      * Constructs a {@code QR} decomposer which computes either the full or reduced {@code QR} decomposition.
59      * @param fullQR Flag for determining if the full {@code QR} decomposition should be used.
60      *               If true, the full {@code QR} decomposition will be computed, if false,
61      *               the reduced {@code QR} decomposition will be computed.
62      */
63     public QRDecomposition(boolean fullQR) {
64         this.fullQR = fullQR;
65     }
66 
67 
68     /**
69      * Applies {@code QR} decomposition to the source matrix.
70      *
71      * @param src The source matrix to decompose. Not modified.
72      * @return A reference to this decomposer.
73      */
74     @Override
75     public QRDecomposition<T> decompose(T src) {
76         if(fullQR) {
77             this.full(src);
78         } else {
79             this.reduced(src);
80         }
81 
82         return this;
83     }
84 
85 
86     /**
87      * Computes the reduced QR decomposition on the src matrix.
88      * @param src The source matrix to decompose.
89      */
90     protected abstract void full(T src);
91 
92 
93     /**
94      * Computes the full QR decomposition on the src matrix.
95      * @param src The source matrix to decompose.
96      */
97     protected abstract void reduced(T src);
98 
99 
100     /**
101      * Gets the {@code Q} matrix from the {@code QR} decomposition.
102      * @return The {@code Q} matrix from the {@code QR} decomposition.
103      */

105         return Q;
106     }
107 
108 
109     /**

112      */
113     public T getR(){
114         return R;
115     }
