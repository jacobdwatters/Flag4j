34 /**
35  * Instances of this class can be used to compute the singular value decomposition (SVD) of a real dense matrix.
36  * That is, decompose a rectangular matrix {@code M} as {@code M=USV<sup>H</sup>} where {@code U} and {@code V} are
37  * unitary matrices whose columns are the left and right singular vectors of {@code M} and {@code S} is a rectangular
38  * diagonal matrix containing the singular values of {@code M}.
39  */
40 public class ComplexSVD extends SingularValueDecomposition<CMatrix> {
41 
42 
43     /**
44      * Creates a decomposer to compute the singular value decomposition of a real matrix. The left and right singular
45      * vectors will be computed.
46      */
47     public ComplexSVD() {
48         super(true);
49     }
50 
51 
52     /**
53      * Creates a decomposer to compute the Schur decomposition.
54      *
55      * @param computeUV A flag which indicates if the unitary matrices {@code Q} and {@code V} should be computed
56      *                  (i.e. the singular vectors).<br>
57      *                  - If true, the {@code Q} and {@code V} matrices will be computed.<br>
58      *                  - If false, the {@code Q} and {@code V} matrices  will <b>not</b> be computed. If it is not needed, this may
59      *                  provide a performance improvement.
60      */
61     public ComplexSVD(boolean computeUV) {
62         super(computeUV);
63     }
64 
65 
66     /**
67      * Applies decomposition to the source matrix.
68      *
69      * @param src The source matrix to decompose.
70      * @return A reference to this decomposer.
71      */
72     @Override
73     public ComplexSVD decompose(CMatrix src) {
74         Vector S1;
75 
76         if(computeUV) {
77             // Compute right singular vectors.
78             CMatrix[] pairs = Eigen.getEigenPairs(src.H().mult(src));
79             V = pairs[1];
80 
81             // Compute left singular vectors.
82             if(src.isSquare()) {
83                 U = src.mult(V);
84             } else if(src.numCols > src.numRows) {
85                 U = src.mult(V).getSlice(0, src.numRows, 0, src.numRows);
86             } else {
87                 U = new CMatrix(src.numRows);
88                 U.setSlice(src.mult(V), 0, 0);
89             }
90 
91             // Compute singular values.
92             S1 = pairs[0].toVector().toReal().abs().sqrt();
93 
94             // Copy singular values to diagonal of S and scale left singular vectors properly.
95         } else {
96             S1 = Eigen.getEigenValues(src.H().mult(src)).toReal().abs().sqrt();
97         }
98 
99         S = new Matrix(src.shape);
100         int stopIdx = Math.min(S.numRows, S.numCols);
101         for(int i=0; i<stopIdx; i++) {
102             S.set(S1.entries[i], i, i);
103 
104             if(computeUV && S1.entries[i] != 0) {
105                 // Properly scale the left singular vectors so that Mv = su.
106                 divCols(i, S1.entries[i]);
107             }
108         }
109 
110         return this;
111     }
112 
113 
114     /**
115      * Divides a specified column of a matrix by a scalar value.
116      * @param colIdx Index of column to divide.
117      * @param scalValue Value to divide column by.
118      */
119     private void divCols(int colIdx, double scalValue) {
120         int idx = colIdx;
121         for(int i=0; i<U.numRows; i++) {
122             U.entries[idx] = U.entries[idx].div(scalValue);
123             idx+=U.numCols;
124         }
125     }
126 }
