33  * <p>This abstract class specifies methods for computing the Cholesky decomposition of a hermation
34  * positive-definite matrix.</p>
35  *
36  * <p>Given a hermation positive-definite matrix {@code A}, the Cholesky decomposition will decompose it into
37  * {@code A=LL<sup>*</sup>} where {@code L} is a lower triangular matrix and {@code L<sup>*</sup>} is the conjugate
38  * transpose of {@code L}.</p>
39  */

44      * Decompose a matrix into {@code A=LL<sup>*</sup>} where {@code L} is a lower triangular matrix and {@code L<sup>*</sup>} is the conjugate
45      * transpose of {@code L}.
46      *
47      * @param src The source matrix to decompose. Must be hermation positive-definite.
48      * @return A reference to this decomposer.
49      * @throws IllegalArgumentException If {@code src} is not hermation positive-definite.
50      */
51     @Override
52     public ComplexCholeskyDecomposition decompose(CMatrix src) {
53         // TODO: Check that matrix is positive definite.
54         ParameterChecks.assertSquare(src.shape);
55 
56         L = new CMatrix(src.numRows);
57         CNumber sum;
58 
59         int lIndex1;
60         int lIndex2;
61         int lIndex3;
62 
63         for(int i=0; i<src.numCols; i++) {
64             lIndex1 = i*L.numCols;
65 
66             for(int j=0; j<=i; j++) {
67                 sum = new CNumber();
68                 lIndex2 = j*L.numCols;
69                 lIndex3 = lIndex1 + j;
70 
71                 for(int k=0; k<j; k++) {
72                     sum.addEq(L.entries[lIndex1 + k].mult(L.entries[lIndex2 + k].conj()));
73                 }
74 
75                 if(i==j) {
76                     L.entries[lIndex3] = CNumber.sqrt(src.entries[lIndex3].sub(sum));
77                 } else {
78                     if(!L.entries[j*(L.numCols + 1)].equals(CNumber.ZERO)) {
79                         L.entries[lIndex3] = (src.entries[lIndex3].sub(sum)).div(L.entries[lIndex2 + j]);
80                     }
81                 }
82             }
83         }
84 
85         return this;
86     }
