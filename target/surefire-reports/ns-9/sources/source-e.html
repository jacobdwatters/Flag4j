33 /**
34  * <p>Instances of this class compute the {@code QR} decomposition of a real dense matrix.</p>
35  * <p>The {@code QR} decomposition, decomposes a matrix {@code A} into an orthogonal matrix {@code Q}
36  * and an upper triangular matrix {@code R} such that {@code A=QR}.</p>
37  */
38 public final class RealQRDecomposition extends QRDecomposition<Matrix> {
39 
40 
41     /**
42      * Constructs a {@code QR} decomposer which computes the full {@code QR} decomposition.
43      */
44     public RealQRDecomposition() {
45         super();
46     }
47 
48 
49     /**
50      * Constructs a {@code QR} decomposer which computes either the full or reduced {@code QR} decomposition.
51      * @param fullQR Flag for determining if the full {@code QR} decomposition should be used.
52      *               If true, the full {@code QR} decomposition will be computed, if false,
53      *               the reduced {@code QR} decomposition will be computed.
54      */
55     public RealQRDecomposition(boolean fullQR) {
56         super(fullQR);
57     }
58 
59 
60     /**
61      * Computes the reduced QR decomposition on the src matrix.
62      * @param src The source matrix to decompose.
63      */
64     @Override
65     protected void reduced(Matrix src) {
66         full(src); // First compute the full decomposition
67 
68         int k = Math.min(src.numRows, src.numCols);
69 
70         // Now reduce the decomposition
71         Q = Q.getSlice(0, src.numRows, 0, k);
72         R = R.getSlice(0, k, 0, src.numCols);
73     }
74 
75 
76     /**
77      * Computes the full QR decomposition on the src matrix.
78      * @param src The source matrix to decompose.
79      */
80     @Override
81     protected void full(Matrix src) {
82         R = new Matrix(src); // Initialize R to the values in src.
83         int m = R.numRows, n = R.numCols;
84         int stop = Math.min(n, m-1);
85 
86         Matrix H;
87         Vector col;
88 
89         // Initialize Q to the identity matrix.
90         Q = Matrix.I(R.numRows);
91 
92         for(int i=0; i<stop; i++) {
93             col = R.getColBelow(i, i).toVector();
94 
95             // If the column has zeros below the diagonal it is in the correct form. No need to compute reflector.
96             if(!col.isZeros()) {
97                 H = Matrix.I(m);
98                 H.setSlice(Householder.getReflector(col), i, i);
99 
100                 Q = Q.mult(H); // Apply Householder reflector to Q
101                 R = H.mult(R); // Apply Householder reflector to R
102             }
103         }
104     }
105 }
