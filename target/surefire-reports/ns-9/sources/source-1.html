30 /**
31  * <p>This abstract class specifies methods for computing the Cholesky decomposition of a positive-definite matrix.</p>
32  *
33  * <p>Given a hermation positive-definite matrix {@code A}, the Cholesky decomposition will decompose it into
34  * {@code A=LL<sup>*</sup>} where {@code L} is a lower triangular matrix and {@code L<sup>*</sup>} is the conjugate
35  * transpose of {@code L}.</p>
36  *
37  * <p>If {@code A} is a real valued symmetric positive-definite matrix, then the decomposition simplifies to
38  * {@code A=LL<sup>T</sup>}.</p>
39  *
40  * @param <T> The type of matrix to compute the Cholesky decomposition of.
41  */
42 public abstract class CholeskyDecomposition<T extends MatrixMixin<T, ?, ?, ?, ?, ?, ?>>
43         implements Decomposition<T> {
44 
45     /**
46      * The lower triangular matrix resulting from the Cholesky decomposition {@code A=LL<sup>*</sup>}.
47      */
48     protected T L;
49 
50 
51     /**
52      * Gets the {@code L} matrix computed by the Cholesky decomposition {@code A=LL<sup>*</sup>}.
53      * @return The {@code L} matrix from the Cholesky decomposition {@code A=LL<sup>*</sup>}.
54      */
55     public T getL() {
56         return L;
57     }
58 
59 
60     /**
61      * Gets the conjugate transpose of the {@code L} matrix computed by the Cholesky decomposition {@code A=LL<sup>*</sup>}.

63      */
64     public abstract T getLH();
65 }
