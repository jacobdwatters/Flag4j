81     /**
82      * Computes the LU decomposition using no pivoting (i.e. rows and columns are not swapped).
83      */
84     @Override
85     protected void noPivot() {
86         P = Q = null; // P and Q are not used here.
87 
88         // Using Gaussian elimination and no pivoting
89         for(int j=0; j<LU.numCols; j++) {
90             if(j<LU.numRows && (LU.entries[j*LU.numCols + j]).magAsDouble() < zeroPivotTol) {
91                 throw new ArithmeticException("Zero pivot encountered in decomposition." +
92                         " Consider using LU decomposition with partial pivoting.");
93             }
94 
95             computeRows(j);
96         }
97     }

100     /**
101      * Computes the LU decomposition using partial pivoting (i.e. row swapping).
102      */
103     @Override
104     protected void partialPivot() {
105         P = Matrix.I(LU.numRows);
106         Q = null; // Q is not used here.
107         int maxIndex;
108 
109         // Using Gaussian elimination with row pivoting.
110         for(int j=0; j<LU.numCols; j++) {
111             maxIndex = maxColIndex(j); // Find row index of max value (in absolute value) in column j so that the index >= j.
112 
113             // Make the appropriate swaps in LU and P (This is the partial pivoting step).
114             if(j!=maxIndex && maxIndex>=0) {
115                 LU.swapRows(j, maxIndex);
116                 P.swapRows(j, maxIndex);
117             }
118 
119             computeRows(j);
120         }
121     }

124     /**
125      * Computes the LU decomposition using full/rook pivoting (i.e. row and column swapping).
126      */
127     @Override
128     protected void fullPivot() {
129         P = Matrix.I(LU.numRows);
130         Q = Matrix.I(LU.numCols);
131         int[] maxIndex;
132 
133         // Using Gaussian elimination with row and column (rook) pivoting.
134         for(int j=0; j<LU.numCols; j++) {
135             maxIndex = maxIndex(j);
136 
137             // Make the appropriate swaps in LU, P and Q (This is the full pivoting step).
138             if(j!=maxIndex[0] && maxIndex[0]!=-1) {
139                 LU.swapRows(j, maxIndex[0]);
140                 P.swapRows(j, maxIndex[0]);
141             }
142             if(j!=maxIndex[1] && maxIndex[1]!=-1) {
143                 LU.swapCols(j, maxIndex[1]);
144                 Q.swapCols(j, maxIndex[1]);
145             }
146 
147             computeRows(j);
148         }
149     }

152     /**
153      * Helper method which computes rows in the gaussian elimination algorithm.
154      * @param j Column for which to compute values to the right of.
155      */
156     private void computeRows(int j) {
157         CNumber m;
158 
159         for(int i=j+1; i<LU.numRows; i++) {
160             m = LU.entries[i*LU.numCols + j];
161             m = LU.entries[j*LU.numCols + j].equals(CNumber.ZERO) ? m.copy() : m.div(LU.entries[j*LU.numCols + j]);
162 
163             for(int k=j; k<LU.numCols; k++) {
164                 LU.entries[i*LU.numCols + k] = LU.entries[i*LU.numCols + k].sub(m.mult(LU.entries[j*LU.numCols + k]));
165                 LU.entries[i*LU.numCols + j] = m;
166             }
167         }
168     }

171     /**
172      * Computes the max absolute value in a column so that the row is >= j.
173      * @param j column index.
174      * @return The index of the maximum absolute value in the specified column such that the row is >= j.
175      */
176     private int maxColIndex(int j) {
177         int maxIndex = -1;
178         double currentMax = -1;
179         double value;
180 
181         for(int i=j; i<LU.numRows; i++) {
182             value = LU.entries[i*LU.numCols+j].magAsDouble();
183             if(value > currentMax) {
184                 currentMax = value;
185                 maxIndex = i;
186             }
187         }
188 
189         return maxIndex;
190     }

193     /**
194      * Computes maximum absolute value in sub portion of LU matrix below and right of {@code startIndex, startIndex}.
195      * @param startIndex Top left index of row and column of sub matrix.
196      * @return Index of maximum absolute value in specified sub matrix.
197      */
198     private int[] maxIndex(int startIndex) {
199         double currentMax = -1;
200         int[] index = {-1, -1};
201         double value;
202         int idx;
203 
204         for(int i=startIndex; i<LU.numRows; i++) {
205             idx = i*LU.numCols;
206 
207             for(int j=startIndex; j<LU.numCols; j++) {
208                 value = LU.entries[idx+j].magAsDouble();
209                 if(value > currentMax) {
210                     currentMax = value;
211                     index[0] = i;
212                     index[1] = j;
213                 }
214             }
215         }
216 
217         return index;
218     }

226     @Override
227     public CMatrix getL() {
228         CMatrix L = new CMatrix(LU.numRows, Math.min(LU.numRows, LU.numCols));
229 
230         // Copy L values from LU matrix.
231         for(int i=0; i<LU.numRows; i++) {
232             if(i<LU.numCols) {
233                 L.entries[i*L.numCols+i] = new CNumber(1); // Set principle diagonal to be ones.
234             }
235 
236             System.arraycopy(LU.entries, i*LU.numCols, L.entries, i*L.numCols, i);
237         }
238 
239         return L;
240     }
