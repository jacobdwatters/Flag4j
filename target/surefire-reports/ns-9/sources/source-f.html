33 /**
34  * Instances of this class can be used to compute the singular value decomposition (SVD) of a real dense matrix.
35  * That is, decompose a rectangular matrix {@code M} as {@code M=USV<sup>T</sup>} where {@code U} and {@code V} are
36  * orthogonal matrices whose columns are the left and right singular vectors of {@code M} and {@code S} is a rectangular
37  * diagonal matrix containing the singular values of {@code M}.
38  */
39 public class RealSVD extends SingularValueDecomposition<Matrix> {
40 
41     // TODO: This is a very rough implementation of SVD. To compute the SVD of a matrix M, the
42     //  matrix M^T*M should not be explicitly computed. Ideally, a modified version of the double shift implicit QR
43     //  algorithm should be used to compute the SVD directly without forming M^T*M explicitly.
44 
45     /**
46      * Creates a decomposer to compute the singular value decomposition of a real matrix. The left and right singular
47      * vectors will be computed.
48      */
49     public RealSVD() {
50         super(true);
51     }
52 
53 
54     /**
55      * Creates a decomposer to compute the singular value decomposition of a real matrix.
56      * @param computeUV A flag which indicates if the orthogonal matrices {@code Q} and {@code V} should be computed
57      *                  (i.e. the singular vectors).<br>
58      *                 - If true, the {@code Q} and {@code V} matrices will be computed.<br>
59      *                 - If false, the {@code Q} and {@code V} matrices  will <b>not</b> be computed. If they are not
60      *                 needed, this may provide a performance improvement.
61      */
62     public RealSVD(boolean computeUV) {
63         super(computeUV);
64     }
65 
66 
67     /**
68      * Applies decomposition to the source matrix.
69      *
70      * @param src The source matrix to decompose.
71      * @return A reference to this decomposer.
72      */
73     @Override
74     public RealSVD decompose(Matrix src) {
75         Vector S1;
76 
77         if(computeUV) {
78             // Compute right singular vectors.
79             CMatrix[] pairs = Eigen.getEigenPairs(src.T().mult(src));
80             V = pairs[1].toReal();
81 
82             // Compute left singular vectors.
83             if(src.isSquare()) {
84                 U = src.mult(V);
85             } else if(src.numCols > src.numRows) {
86                 U = src.mult(V).getSlice(0, src.numRows, 0, src.numRows);
87             } else {
88                 U = new Matrix(src.numRows);
89                 U.setSlice(src.mult(V), 0, 0);
90             }
91 
92             // Compute singular values.
93             S1 = pairs[0].toVector().toReal().abs().sqrt();
94 
95             // Copy singular values to diagonal of S and scale left singular vectors properly.
96         } else {
97             S1 = Eigen.getEigenValues(src.T().mult(src)).toReal().abs().sqrt();
98         }
99 
100         S = new Matrix(src.shape);
101         int stopIdx = Math.min(S.numRows, S.numCols);
102         for(int i=0; i<stopIdx; i++) {
103             S.set(S1.entries[i], i, i);
104 
105             if(computeUV && S1.entries[i] != 0) {
106                 // Properly scale the left singular vectors so that Mv = su.
107                 divCols(i, S1.entries[i]);
108             }
109         }
110 
111         return this;
112     }
113 
114 
115     /**
116      * Divides a specified column of the {@code U} matrix in the SVD by a scalar value.
117      * @param colIdx Index of column to divide.
118      * @param scalValue Value to divide column by.
119      */
120     private void divCols(int colIdx, double scalValue) {
121         int idx = colIdx;
122         for(int i=0; i<U.numRows; i++) {
123             U.entries[idx] /= scalValue;
124             idx+=U.numCols;
125         }
126     }
127 }
