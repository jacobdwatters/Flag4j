65     /**

67      * @param src1 Entries of first tensor.
68      * @param shape1 Shape of first tensor.
69      * @param src2 Entries of second tensor.
70      * @param shape2 Shape of second tensor.

72      */
73     private static boolean tensorEquals(double[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
