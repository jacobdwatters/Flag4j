46     /**
47      * Computes the matrix multiplication between two sparse matrices using a standard algorithm.
48      * @param src1 Non-zero entries of the first sparse matrix.
49      * @param rowIndices1 Row indices of non-zero entries for the first sparse matrix.
50      * @param colIndices1 Column indices of non-zero entries for the first sparse matrix.
51      * @param shape1 Shape of the first sparse matrix.
52      * @param src2 Non-zero entries of the second sparse matrix.
53      * @param rowIndices2 Row indices of non-zero entries for the second sparse matrix.
54      * @param colIndices2 column indices of non-zero entries for the second sparse matrix.
55      * @param shape2 Shape of the second sparse matrix.
56      * @return The result of the matrix multiplication stored in a dense matrix.
57      */
58     public static CNumber[] standard(CNumber[] src1, int[] rowIndices1, int[] colIndices1, Shape shape1,
59                                      CNumber[] src2, int[] rowIndices2, int[] colIndices2, Shape shape2) {
60 
61         int rows1 = shape1.dims[Axis2D.row()];
62         int cols2 = shape2.dims[Axis2D.col()];
63 
64         CNumber[] dest = new CNumber[rows1*cols2];
65         ArrayUtils.fill(dest, CNumber.ZERO);
66 
67         // r1, c1, r2, and c2 store row/column indices for non-zero values in src1 and src2.
68         int r1, c1, r2, c2;
69 
70         for(int i=0; i<src1.length; i++) {
71             r1 = rowIndices1[i]; // = i
72             c1 = colIndices1[i]; // = k
73 
74             for(int j=0; j<src2.length; j++) {
75                 r2 = rowIndices2[j]; // = k
76                 c2 = colIndices2[j]; // = j
77 
78                 if(c1==r2) { // Then we multiply and add to sum.
79                     dest[r1*cols2 + c2].addEq(src1[i].mult(src2[j]));
80                 }
81             }
82         }
83 
84         return dest;
85     }
