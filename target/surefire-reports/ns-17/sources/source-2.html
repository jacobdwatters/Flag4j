48      * Computes the matrix multiplication between two sparse matrices using a standard algorithm.
49      * @param src1 Non-zero entries of the first sparse matrix.
50      * @param rowIndices1 Row indices of non-zero entries for the first sparse matrix.
51      * @param colIndices1 Column indices of non-zero entries for the first sparse matrix.
52      * @param shape1 Shape of the first sparse matrix.
53      * @param src2 Non-zero entries of the second sparse matrix.
54      * @param rowIndices2 Row indices of non-zero entries for the second sparse matrix.
55      * @param colIndices2 column indices of non-zero entries for the second sparse matrix.
56      * @param shape2 Shape of the second sparse matrix.
57      * @return The result of the matrix multiplication stored in a dense matrix.
58      */
59     public static CNumber[] standard(CNumber[] src1, int[] rowIndices1, int[] colIndices1, Shape shape1,
60                                      double[] src2, int[] rowIndices2, int[] colIndices2, Shape shape2) {
61 
62         int rows1 = shape1.dims[Axis2D.row()];
63         int cols2 = shape2.dims[Axis2D.col()];
64 
65         CNumber[] dest = new CNumber[rows1*cols2];
66 
67         // r1, c1, r2, and c2 store row/column indices for non-zero values in src1 and src2.
68         int r1, c1, r2, c2;
69 
70         for(int i=0; i<src1.length; i++) {
71             r1 = rowIndices1[i]; // = i
72             c1 = colIndices1[i]; // = k
73 
74             for(int j=0; j<src2.length; j++) {
75                 r2 = rowIndices2[j]; // = k
76                 c2 = colIndices2[j]; // = j
77 
78                 if(c1==r2) { // Then we multiply and add to sum.
79                     dest[r1*cols2 + c2].addEq(src1[i].mult(src2[j]));
80                 }
81             }
82         }
83 
84         return dest;
85     }
