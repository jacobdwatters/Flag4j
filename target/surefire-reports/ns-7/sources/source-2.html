30 /**
31  * <p>This abstract class specifies methods for computing the {@code QR} decomposition of a matrix.</p>
32  * <p>The {@code QR} decomposition, decomposes a matrix {@code A} into an orthogonal matrix {@code Q}
33  * and an upper triangular matrix {@code R} such that {@code A=QR}.</p>
34  */
35 public abstract class QRDecomposition<T extends MatrixBase<?>> implements Decomposition<T> {
36 
37     /**
38      * Storage matrix for {@code Q}.
39      */
40     protected T Q;
41     /**
42      * Storage matrix for {@code R}.
43      */
44     protected T R;
45     /**
46      * Flag for determining if the full {@code QR} decomposition should be used. If true, the full {@code QR} decomposition will be computed,
47      * if false, the reduced QR decomposition will be computed.
48      */
49     protected final boolean fullQR;
50 
51     /**
52      * Constructs {@code QR} decomposer which computes the full {@code QR} decomposition.
53      */
54     public QRDecomposition() {
55         fullQR = true;
56     }
57 
58     /**
59      * Constructs a {@code QR} decomposer which computes either the full or reduced {@code QR} decomposition.
60      * @param fullQR Flag for determining if the full {@code QR} decomposition should be used.
61      *               If true, the full {@code QR} decomposition will be computed, if false,
62      *               the reduced {@code QR} decomposition will be computed.
63      */
64     public QRDecomposition(boolean fullQR) {
65         this.fullQR = fullQR;
66     }
67 
68 
69     /**
70      * Gets the {@code Q} matrix from the {@code QR} decomposition.
71      * @return The {@code Q} matrix from the {@code QR} decomposition.
72      */
73     public T getQ(){
74         return Q;
75     }
76 
77

83         return R;
84     }
