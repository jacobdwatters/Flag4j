37 /**
38  * A {@code TensorInputStream} obtains bytes from a system file using a {@link FileInputStream} containing a serialized
39  * tensor, matrix, or vector. Then, deserializes that object using an {@link ObjectInputStream}.
40  */
41 public class TensorInputStream extends InputStream {
42 
43     /**
44      * Stream for loading bytes from a system file.
45      */
46     private final FileInputStream fileIn;
47     /**
48      * Stream for deserializing an object.
49      */
50     private final ObjectInputStream objectIn;
51 
52 
53     /**
54      * Creates {@link TensorInputStream} which will open a connection with a file through an {@link ObjectInputStream}
55      * and a {@link FileInputStream}. This {@link TensorInputStream} can be used to deserialize a tensor,
56      * matrix, or vector which is stored in a system file.
57      * @param fileName File name, including extension, of tensor, matrix, or vector to read.
58      * @throws IOException If an I/O error occurs while reading the stream header.
59      */
60     public TensorInputStream(String fileName) throws IOException {
61         fileIn = new FileInputStream(fileName);
62         objectIn = new ObjectInputStream(fileIn);
63     }
64 
65 
66     /**
67      * Reads a serialized {@link Tensor tesnor} from a file.
68      * @return The deserialized {@link Tensor tensor object} stored in the specified file.
69      */
70     public Tensor readTensor() throws IOException, ClassNotFoundException {
71         // Deserialize tensor.
72         return (Tensor) objectIn.readObject();
73     }
74

76     /**

78      * @return The deserialized {@link Matrix matrix object} stored in the specified file.

85

88      * @return The deserialized {@link Vector vector object} stored in the specified file.
89      */
