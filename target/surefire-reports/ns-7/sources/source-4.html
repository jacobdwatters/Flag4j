33  * Computes the QR decomposition for a real matrix.
34  */

47      * Constructs a {@code QR} decomposer which computes either the full or reduced {@code QR} decomposition.

58      * Applies decomposition to the source matrix.

61      */
62     @Override

72     /**
73      * Computes the reduced QR decomposition on the src matrix.
74      * @param A The source matrix to decompose.
75      */
76     private void reduced(Matrix A) {
77         full(A); // First compute the full decomposition
78 
79         int k = Math.min(A.numRows, A.numCols);
80 
81         // Now reduce the decomposition
82         Q = Q.getSlice(0, A.numRows, 0, k);
83         R = R.getSlice(0, k, 0, A.numCols);
84     }

87     /**
88      * Computes the full QR decomposition on the src matrix.
89      * @param A The source matrix to decompose.
90      */
91     private void full(Matrix A) {
92         R = new Matrix(A); // Initialize R to the values in A.
93         int m = R.numRows, n = R.numCols;
94         int stop = Math.min(n, m-1);
95 
96         Matrix H, col;
97 
98         // Initialize Q to the identity matrix.
99         Q = Matrix.I(R.numRows);
100 
101         for(int i=0; i<stop; i++) {
102             H = Matrix.I(m);
103             col = R.getColBelow(i, i);
104 
105             if(!col.isZeros()) { // Then a householder transform must be applied
106                 H.setSlice(getHouseholder(col), i, i);
107 
108                 Q = Q.mult(H); // Apply Householder reflector to Q
109                 R = H.mult(R); // Apply Householder reflector to R
110             }
111         }
112     }
113

115     /**
116      * Computes the Householder reflector for the specified column vector.
117      * @param col Column vector to compute Householder reflector for.
118      * @return The Householder transformation matrix.
119      */
120     private Matrix getHouseholder(Matrix col) {
121         Matrix H = Matrix.I(col.numRows);
122         Vector v = col.toVector();
123 
124         double signedNorm = -Math.copySign(v.norm(), v.entries[0]);
125         v = v.scalDiv(v.entries[0] + signedNorm);
126         v.entries[0] = 1;
127 
128         // Create projection matrix
129         Matrix P = v.outer(v).mult(2/v.inner(v));
130         H.subEq(P);
131 
132         return H;
133     }
