35 /**
36  * A class for writing tensors to a file in various formats.
37  */
38 public class TensorOutputStream extends OutputStream {
39 
40     /**
41      * Stream for writing data to a file.
42      */
43     private final FileOutputStream fileOut;
44     /**
45      * Stream for serializing object.
46      */
47     private final ObjectOutputStream objectOut;
48 
49 
50     /**
51      * Creates {@link TensorOutputStream} which will use an {@link ObjectOutputStream} to write serialized data to a
52      * {@link FileOutputStream} which will in-turn write the serialized data to a file.
53      * @param fileName File name, including extension, of file to write data to.
54      * @throws IOException If an I/O error occurs while reading the stream header.
55      */
56     public TensorOutputStream(String fileName) throws IOException {
57         fileOut = new FileOutputStream(fileName);
58         objectOut = new ObjectOutputStream(fileOut);
59     }
60 
61 
62     /**
63      * Writes a {@link TensorBase} object to a file by serializing the object.
64      * @param A Object to write to file. This may be a real/complex dense or sparse tensor, matrix, or vector.
65      */
66     public void write(TensorBase<?, ?, ?, ?, ?, ?, ?> A) throws IOException {
67         objectOut.writeObject(A);
68     }
69 
70 
71     /**
72      * Writes a bite to the output stream. This method will forward the write to {@link ObjectOutputStream#write(int)}.
73      *
74      * @param b the {@code byte}.
75      * @throws IOException if an I/O error occurs. In particular,
76      *                     an {@code IOException} may be thrown if the
77      *                     output stream has been closed.
78      */
79     @Override
80     public void write(int b) throws IOException {
81         objectOut.write(b);
82     }
83 
84 
85     /**
86      * Closes the stream. This method must be called to release any resources associated with the stream.
87      * @throws IOException If an I/O exception has occurred.
88      */
89     @Override
90     public void close() throws IOException {
91         objectOut.close();
92         fileOut.close();
93     }
