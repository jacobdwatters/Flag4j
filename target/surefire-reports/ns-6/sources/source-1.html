39 /**

49     /**
50      * Creates a sparse tensor with specified shape.
51      * @param shape Shape of this tensor.
52      * @param nonZeroEntries Number of non-zero entries in the sparse tensor.
53      * @param entries Non-zero entries of this sparse tensor.
54      * @param indices Indices of the non-zero entries.
55      * @throws IllegalArgumentException If the number of rows in the indices array is not equal to the number of
56      * elements in the entries array.
57      * @throws IllegalArgumentException If the number of columns in the entries array is not equal to the rank of this
58      * tensor.
59      */
60     public ComplexSparseTensorBase(Shape shape, int nonZeroEntries, CNumber[] entries, int[][] indices) {
61         super(shape, nonZeroEntries, entries, indices);
62 
63         if(super.totalEntries().compareTo(BigInteger.valueOf(nonZeroEntries)) < 0) {
64             throw new IllegalArgumentException(ErrorMessages.shapeEntriesError(shape, nonZeroEntries));
65         }
66         ParameterChecks.assertArrayLengthsEq(nonZeroEntries, indices.length);
67     }
