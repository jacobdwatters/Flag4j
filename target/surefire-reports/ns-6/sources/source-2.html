39 /**

42  * @param <U> Dense Tensor type.

50     /**
51      * Creates a sparse tensor with specified shape.
52      * @param shape Shape of this tensor.
53      * @param nonZeroEntries Number of non-zero entries in the sparse tensor.
54      * @param entries Non-zero entries of this sparse tensor.
55      * @param indices Indices of the non-zero entries.
56      * @throws IllegalArgumentException If the number of rows in the indices array is not equal to the number of
57      * elements in the entries array.
58      * @throws IllegalArgumentException If the number of columns in the entries array is not equal to the rank of this
59      * tensor.
60      */
61     public RealSparseTensorBase(Shape shape, int nonZeroEntries, double[] entries, int[][] indices) {
62         super(shape, nonZeroEntries, entries, indices);
63 
64         if(super.totalEntries().compareTo(BigInteger.valueOf(nonZeroEntries)) < 0) {
65             throw new IllegalArgumentException(ErrorMessages.shapeEntriesError(shape, nonZeroEntries));
66         }
67 
68         ParameterChecks.assertArrayLengthsEq(nonZeroEntries, indices.length);
69     }
