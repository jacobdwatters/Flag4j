49      * Checks if two real dense matrices are equal.
50      * @param A First matrix.
51      * @param B Second matrix.
52      * @return True if the two matrices are element-wise equivalent.
53      */
54     public static boolean matrixEquals(CMatrix A, SparseCMatrix B) {
55         boolean equal = true;
56 
57         if(A.shape.equals(B.shape)) {
58             CNumber[] entriesCopy = Arrays.copyOf(A.entries, A.entries.length);
59 
60             int rowIndex, colIndex;
61             int entriesIndex;
62 
63             // Remove all nonZero entries from the entries of this matrix.
64             for(int i=0; i<B.nonZeroEntries(); i++) {
65                 rowIndex = B.rowIndices[i];
66                 colIndex = B.colIndices[i];
67                 entriesIndex = A.shape.entriesIndex(rowIndex, colIndex);
68 
69                 if(!entriesCopy[entriesIndex].equals(B.entries[i])) {
70                     equal = false;
71                     break;
72                 }
73 
74                 entriesCopy[A.shape.entriesIndex(rowIndex, colIndex)] = new CNumber();
75             }
76 
77             if(equal) {
78                 // Now, if this matrix is equal to the sparse matrix, there should only be zeros left in the entriesStack
79                 equal = ArrayUtils.isZeros(entriesCopy);
80             }
81 
82         } else {
83             equal = false;
84         }
85
