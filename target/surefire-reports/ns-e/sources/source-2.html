126     /**
127      * Rounds the values of a tensor with specified precision. Note, if precision is zero, {@link #round(double[])} is
128      * preferred.
129      * @param src Entries of the tensor to round.
130      * @param precision Precision to round to (i.e. the number of decimal places).
131      * @return The result of rounding all entries of the source tensor with the specified precision.
132      * @throws IllegalArgumentException If {@code precision} is negative.
133      */
134     public static double[] round(double[] src, int precision) {
135         if(precision<0) {
136             throw new IllegalArgumentException(ErrorMessages.negValueErr(precision));
137         }
138 
139         BigDecimal bd;
140         double[] dest = new double[src.length];
141 
142         for(int i=0; i<dest.length; i++) {
143             bd = new BigDecimal(Double.toString(src[i]));
144             bd = bd.setScale(precision, RoundingMode.HALF_UP);
145             dest[i] = bd.doubleValue();
146         }
147 
148         return dest;
149     }

153      * Rounds values which are close to zero in absolute value to zero.
154      *
155      * @param threshold Threshold for rounding values to zero. That is, if a value in this tensor is less than
156      *                  the threshold in absolute value then it will be rounded to zero. This value must be non-negative.
157      * @return A copy of this matrix with rounded values.
158      * @throws IllegalArgumentException If {@code threshold} is negative.
159      */
160     public static double[] roundToZero(double[] src, double threshold) {
161         if(threshold<0) {
162             throw new IllegalArgumentException(ErrorMessages.negValueErr(threshold));
163         }
164 
165         double[] dest = new double[src.length];
166 
167         for(int i=0; i<dest.length; i++) {
168             if(Math.abs(src[i]) < threshold) {
169                 dest[i] = 0;

172             }
173         }

175         return dest;
