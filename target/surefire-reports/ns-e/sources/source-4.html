44      * @param src Entries of the vector (or non-zero entries if vector is sparse).
45      * @return The 2-norm of the vector.

76     /**
77      * Computes the {@code p}-norm of a vector.
78      * @param src Entries of the vector (or non-zero entries if vector is sparse).
79      * @param p The {@code p} value in the {@code p}-norm. <br>
80      *          - If {@code p} is {@link Double#POSITIVE_INFINITY}, then this method computes the maximum/infinite norm. <br>
81      *          - If {@code p} is {@link Double#NEGATIVE_INFINITY}, then this method computes the minimum norm. <br>
82      *          Warning, if {@code p} is large in absolute value, overflow errors may occur.
83      * @return The {@code p}-norm of the vector.
84      */
85     public static double norm(double[] src, double p) {
86         if(Double.isInfinite(p)) {
87             if(p > 0) {
88                 return AggregateReal.maxAbs(src); // Maximum / infinite norm.
89             } else {
90                 return AggregateReal.minAbs(src); // Minimum norm.
91             }
92         } else {
93             double norm = 0;
94 
95             for(double v : src) {
96                 norm += Math.pow(Math.abs(v), p);
97             }
98 
99             return Math.pow(norm, 1.0/p);
100         }
101     }
