59     /**
60      * Computes the reduced row echelon form of a matrix. For a non-reduced row echelon form see {@link #ref(Matrix)}.
61      * @param A The matrix for which to compute the reduced row echelon form.
62      * @return A matrix in reduced row echelon form which is row-equivalent to this matrix.
63      */
64     public static Matrix rref(Matrix A) {
65         Matrix U = new RealLUDecomposition().decompose(A).getU();
66         int colStop = Math.min(U.numCols, U.numRows);
67         int pivotRow;
68         int iRow;
69         double m;
70         double pivotValue;
71 
72         for(int j=0; j<colStop; j++) {
73             pivotRow = j*U.numCols;
74             pivotValue = U.entries[pivotRow + j];
75 
76             // Scale row so pivot is 1.
77             for(int k=j; k<U.numCols; k++) {
78                 U.entries[pivotRow + k] = U.entries[pivotRow + k]/pivotValue;
79             }
80 
81             // Zero out column above pivot.
82             for(int i=0; i<j; i++) {
83                 iRow = i*U.numCols;
84                 m = U.entries[iRow + j];
85 
86                 for(int k=j; k<U.numCols; k++) {
87                     U.entries[iRow + k] -= m*U.entries[pivotRow + k];
88                 }
89             }
90         }
91 
92         return U;
93     }

96     /**
97      * Computes the reduced row echelon form of a matrix. For a non-reduced row echelon form see {@link #ref(CMatrix)}.
98      * @param A The matrix for which to compute the reduced row echelon form.
99      * @return A matrix in reduced row echelon form which is row-equivalent to this matrix.
100      */
101     public static CMatrix rref(CMatrix A) {
102         CMatrix U = new ComplexLUDecomposition().decompose(A).getU();
103         int colStop = Math.min(U.numCols, U.numRows);
104         int pivotRow;
105         int iRow;
106         CNumber m;
107 
108         for(int j=0; j<colStop; j++) {
109             pivotRow = j*U.numCols;
110             m = U.entries[pivotRow + j];
111 
112             // Scale row so pivot is 1.
113             for(int k=j; k<U.numCols; k++) {
114                 U.entries[pivotRow + k] = U.entries[pivotRow + k].div(m);
115             }
116 
117             // Zero out column above pivot.
118             for(int i=0; i<j; i++) {
119                 iRow = i*U.numCols;
120                 m = U.entries[iRow + j];
121 
122                 for(int k=j; k<U.numCols; k++) {
123                     U.entries[iRow + k].subEq(m.mult(U.entries[pivotRow + k]));
124                 }
125             }
126         }
127 
128         return U;
129     }
