43      * Computes the eigenvalues of a 2x2 matrix explicitly.
44      * @param src The 2x2 matrix to compute the eigenvalues of.
45      * @return A complex vector containing the eigenvalues of the 2x2 {@code src} matrix.
46      */
47     public static CVector get2x2EigenValues(Matrix src) {
48         // TODO: While theoretically correct, there are some numerical issues here.
49         CVector lambda = new CVector(2);
50 
51         // Get the four entries from lower right 2x2 sub-matrix.
52         double a = src.entries[0];
53         double b = src.entries[1];
54         double c = src.entries[2];
55         double d = src.entries[3];
56 
57         double det = a*d - b*c; // 2x2 determinant.
58         double htr = (a+d)/2; // Half of the 2x2 trace.
59 
60         // 2x2 block eigenvalues.
61         lambda.entries[0] = CNumber.sqrt(CNumber.pow(htr, 2).sub(det)).add(htr);
62         lambda.entries[1] = CNumber.sqrt(CNumber.pow(htr, 2).sub(det)).add(htr);
63 
64         return lambda;
65     }

143      * Computes the eigenvectors of a square real dense matrix.
144      * @param src The matrix to compute the eigenvectors of.
145      * @return A matrix containing the eigenvectors of {@code src} as its columns.
146      */
147     public static CMatrix getEigenVectors(Matrix src) {
148         SchurDecomposition<Matrix> schur = new RealSchurDecomposition(true).decompose(src);
149         CMatrix U = schur.getU();
150 
151         if(src.isSymmetric()) {
152             // Then the columns of U are the complete orthonormal set of eigenvectors of the src matrix.
153             return U;
154         } else {
155             // For a non-symmetric matrix, only the first column of U will be an eigenvector of the src matrix.
156             CMatrix Q = getEigenVectorsTriu(schur.getT()); // Compute the eigenvectors of T.
157             return U.mult(Q); // Convert the eigenvectors of T to the eigenvectors of the src matrix.
158         }
159     }

163      * Computes the eigenvectors of an upper triangular matrix.
164      * @param T The upper triangular matrix to compute the eigenvectors of.
165      * @return A matrix containing the eigenvectors of {@code T} as its columns.
166      */
167     public static CMatrix getEigenVectorsTriu(CMatrix T) {
168         ComplexBackSolver backSolver = new ComplexBackSolver();
169         Matrix I = Matrix.I(T.numRows);
170         CMatrix Q = new CMatrix(T.numRows);
171 
172         for(int j=0; j<T.numRows; j++) {
173             CMatrix t = I.mult(T.get(j, j));
174             CMatrix S = T.sub(t).getSlice(0, j+1, 0, j+1);
175             CMatrix S_hat = S.getSlice(0, j, 0, j);
176             CVector r = S.getSlice(0, j, S.numCols-1, S.numCols).toVector();
177             CVector v;
178 
179             if(S_hat.entries.length > 0) {
180                 v = backSolver.solve(S_hat, r.mult(-1));
181                 v = v.join(new CVector(1.0));
182             } else {
183                 v = new CVector(1.0);
184             }
185 
186             v = v.normalize().join(new Vector(T.numRows-v.size));
187             Q.setCol(v, j);
188         }
189 
190         return Q;
191     }
