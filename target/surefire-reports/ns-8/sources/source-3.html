166      * @param src The matrix to compute the eigenvectors of.
167      * @return A matrix containing the eigenvectors of {@code src} as its columns.
168      */
169     public static CMatrix getEigenVectors(Matrix src) {
170         SchurDecomposition<Matrix> schur = new RealSchurDecomposition(true).decompose(src);
171         CMatrix U = schur.getU();
172 
173         if(src.isSymmetric()) {
174             // Then the columns of U are the complete orthonormal set of eigenvectors of the src matrix.
175             return U;
176         } else {
177             // For a non-symmetric matrix, only the first column of U will be an eigenvector of the src matrix.
178             CMatrix Q = getEigenVectorsTriu(schur.getT()); // Compute the eigenvectors of T.
179             return U.mult(Q); // Convert the eigenvectors of T to the eigenvectors of the src matrix.
180         }

205      * Computes the eigenvectors of an upper triangular matrix.
206      * @param T The upper triangular matrix to compute the eigenvectors of.
207      * @return A matrix containing the eigenvectors of {@code T} as its columns.
208      */
209     public static CMatrix getEigenVectorsTriu(Matrix T) {
210         RealBackSolver backSolver = new RealBackSolver();
211         Matrix I = Matrix.I(T.numRows);
212         CMatrix Q = new CMatrix(T.numRows);
213 
214         for(int j=0; j<T.numRows; j++) {
215             Matrix t = I.mult(T.get(j, j)); // TODO: add diag(...) method so this scalar multiplication doesnt need to be computed.
216             Matrix S = T.sub(t).getSlice(0, j+1, 0, j+1);
217             Matrix S_hat = S.getSlice(0, j, 0, j);
218             Vector r = S.getSlice(0, j, S.numCols-1, S.numCols).toVector();
219             Vector v;
220 
221             if(S_hat.entries.length > 0) {
222                 v = backSolver.solve(S_hat, r.mult(-1));
223                 v = v.join(new Vector(1.0));
224             } else {
225                 v = new Vector(1.0);
226             }
227 
228             v = v.normalize().join(new Vector(T.numRows-v.size));
229             Q.setCol(v.entries, j);
230         }
231 
232         return Q;
233     }
