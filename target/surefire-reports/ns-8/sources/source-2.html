44     /**
45      * <p>Computes the {@link RealLUDecomposition LU factorization} of a matrix using partial pivoting. That is, decomposes an m-by-n matrix {@code A} into {@code PA=LU}
46      * where {@code P} is a permutation matrix, {@code L} is a unit lower triangular matrix, and U is an upper triangular matrix.</p>
47      * @param A Matrix to decompose.
48      * @return Returns an array of matrices containing in order {@code {P, L, U}} corresponding to {@code PA=LU}.
49      */
50     public static Matrix[] lu(Matrix A) {
51         RealLUDecomposition LU = new RealLUDecomposition();
52         LU.decompose(A);
53         return new Matrix[]{LU.getP(), LU.getL(), LU.getU()};
54     }

57     /**
58      * <p>Computes the {@link ComplexLUDecomposition LU factorization} of a complex matrix using partial pivoting.
59      * That is, decomposes an m-by-n matrix {@code A} into {@code PA=LU} where {@code P} is a permutation matrix,
60      * {@code L} is a unit lower triangular matrix, and U is an upper triangular matrix.</p>
61      * @param A Matrix to decompose.
62      * @return Returns an array of matrices containing in order {@code {P, L, U}} corresponding to {@code PA=LU}. Note,
63      * although {@code P} is always real, it will be wrapped in a complex Matrix.
64      */
65     public static CMatrix[] lu(CMatrix A) {
66         ComplexLUDecomposition LU = new ComplexLUDecomposition();
67         LU.decompose(A);
68         return new CMatrix[]{LU.getP().toComplex(), LU.getL(), LU.getU()};
69     }
