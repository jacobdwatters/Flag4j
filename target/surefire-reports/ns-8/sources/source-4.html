51      * Inverts an upper triangular matrix. <b>WARNING:</b> this method does not check that the matrix is actually
52      * upper triangular.
53      * @param src Upper triangular matrix to compute the inverse of.
54      * @return The inverse of the upper triangular matrix.
55      * @throws RuntimeException If the matrix is singular (i.e. has at least one zero along the diagonal).
56      * @throws IllegalArgumentException If the matrix is not square.
57      */
58     public static Matrix invTriU(Matrix src) {
59         ParameterChecks.assertSquare(src.shape);
60         if(zeroOnDiag(src)) {
61             throw new RuntimeException("Cannot invert, matrix is singular.");
62         }
63 
64         RealBackSolver backSolver = new RealBackSolver();
65         Matrix I = Matrix.I(src.shape);
66         Matrix inverse = new Matrix(src.shape);
67 
68         for(int i=0; i<I.numCols; i++) {
69             Vector col = backSolver.solve(src, I.getColAsVector(i));
70             inverse.setCol(col.entries, i);
71         }
72 
73         return inverse;
74     }

114

116 
117         double value;

127             }
