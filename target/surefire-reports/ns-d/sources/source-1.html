51      * Computes the element-wise division of two tensors.
52      * @param src1 First tensor in element-wise division.
53      * @param shape1 Shape of the first tensor.
54      * @param src2 Second tensor in element-wise division.
55      * @param shape2 Shape of the second tensor.

57      * @throws IllegalArgumentException If the tensors do not have the same shape.

59     public static CNumber[] elemDiv(CNumber[] src1, Shape shape1, double[] src2, Shape shape2) {
60         ParameterChecks.assertEqualShape(shape1, shape2);

64             product[i] = src1[i].div(src2[i]);

92     /**
93      * Computes the element-wise division of two tensors.
94      * @param src1 First tensor in element-wise division.
95      * @param shape1 Shape of the first tensor.
96      * @param src2 Second tensor in element-wise division.
97      * @param shape2 Shape of the second tensor.
98      * @return The element-wise division of the two tensors.
99      * @throws IllegalArgumentException If the tensors do not have the same shape.
100      */
101     public static CNumber[] elemDiv(double[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
102         ParameterChecks.assertEqualShape(shape1, shape2);
103         CNumber[] quotient = new CNumber[src1.length];
104         double divisor;
105 
106         for(int i=0; i<quotient.length; i++) {
107             divisor = src2[i].re*src2[i].re + src2[i].im*src2[i].im;
108             quotient[i] = new CNumber(src1[i]*src2[i].re / divisor, -src1[i]*src2[i].im / divisor);
109         }
110 
111         return quotient;
112     }

138      * Chooses if a concurrent algorithm for element-wise multiplication should be used based on the shape of the two tensors.
