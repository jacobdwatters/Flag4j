557      * Dynamically chooses matrix multiply algorithm based on the shapes of the two matrices to multiply.
558      * @param shape1 The shape of the first matrix.
559      * @param shape2 The shape fo the second matrix.
560      * @return The algorithm to use in the matrix multiplication.
561      */
562     public static AlgorithmName chooseAlgorithmComplex(Shape shape1, Shape shape2) {
563         AlgorithmName algorithm;
564 
565         int rows1 = shape1.get(Axis2D.row());
566         int cols1 = shape1.get(Axis2D.col());
567 
568         // TODO: Extract constants to final variables
569         if(getRatio(shape1) >= SQUARENESS_RATIO) {
570             // Then the first matrix is approximately square.
571             if(rows1<=30) {
572                 algorithm = AlgorithmName.REORDERED;
573             } else if(rows1<=250) {
574                 algorithm = AlgorithmName.CONCURRENT_REORDERED;
575             } else {
576             /* For large matrices, use a concurrent, blocked algorithm with the j-k loops swapped for
577             better cache performance on modern systems */
578                 algorithm = AlgorithmName.CONCURRENT_BLOCKED_REORDERED;
579             }
580 
581         } else if(rows1>cols1) {
582             // Then there are more rows than columns in the first matrix
583             if(rows1<=100) {
584                 if(cols1<=4) algorithm = AlgorithmName.REORDERED;
585                 else algorithm = AlgorithmName.CONCURRENT_REORDERED;
586             } else {
587                 if(cols1<=45) algorithm = AlgorithmName.CONCURRENT_REORDERED;
588                 else algorithm = AlgorithmName.CONCURRENT_BLOCKED_REORDERED;
589             }
590         } else {
