48      * Transposes tensor along specified axes using a standard transpose algorithm. In this context, transposing a
49      * tensor is equivalent to swapping a pair of axes.
50      * @param src Entries of the tensor.
51      * @param shape Shape of the tensor to transpose.
52      * @param axis1 First axis to swap in transpose.
53      * @param axis2 Second axis to swap in transpose.
54      * @return The transpose of the tensor along the specified axes.
55      */
56     public static double[] standard(final double[] src, final Shape shape, final int axis1, final int axis2) {
57         if(shape.getRank() < 2) { // Can't transpose tensor with less than 2 axes.
58             throw new IllegalArgumentException("Tensor transpose not defined for rank " + shape.getRank() +
59                     " tensor.");
60         }
61 
62         double[] dest = new double[shape.totalEntries().intValue()];
63         Shape destShape = shape.copy().swapAxes(axis1, axis2);
64         int[] destIndices;
65 
66         for(int i=0; i<src.length; i++) {
67             destIndices = shape.getIndices(i);
68             ArrayUtils.swap(destIndices, axis1, axis2); // Compute destination indices.
69             dest[destShape.entriesIndex(destIndices)] = src[i]; // Apply transpose for the element
70         }
71 
72         return dest;
73     }

167      * Transposes a matrix using the standard algorithm.
168      * @param src Entries of the matrix to transpose.
169      * @param numRows Number of rows in the matrix.
170      * @param numCols Number of columns in the matrix.
171      * @return The transpose of the matrix.
172      */
173     public static double[] standardMatrix(final double[] src, final int numRows, final int numCols) {
174         double[] dest = new double[numRows*numCols];
175 
176         int destIndex, srcIndex, end;
177 
178         for (int i=0; i<numCols; i++) {
179             srcIndex = i;
180             destIndex = i*numRows;
181             end = destIndex + numRows;
182 
183             while (destIndex < end) {
184                 dest[destIndex++] = src[srcIndex];
185                 srcIndex += numCols;
186             }
187         }
188 
189         return dest;
190     }

194      * Transposes a matrix using a blocked algorithm. To get or set the block size see
195      * {@link Configurations#getBlockSize()} or {@link Configurations#setBlockSize(int)}.
196      * @param src Source entries to compute transpose of.
197      * @param numRows Number of rows in the matrix.
198      * @param numCols Number of columns in the matrix.
199      * @return The transpose of this tensor along specified axes
200      */
201     public static double[] blockedMatrix(final double[] src, final int numRows, final int numCols) {
202         double[] dest = new double[numRows*numCols];
203         final int blockSize = Configurations.getBlockSize();
204         int blockRowEnd;
205         int blockColEnd;
206         int srcIndex, destIndex, end;
207 
208         for(int i=0; i<numCols; i+=blockSize) {
209             for(int j=0; j<numRows; j+=blockSize) {
210                 blockRowEnd = Math.min(blockSize, numRows);
211                 blockColEnd = Math.min(i+blockSize, numCols);
212 
213                 // Transpose the block beginning at (i, j)
214                 for(int blockI=i; blockI<blockColEnd; blockI++) {
215                     srcIndex = blockI;
216                     destIndex = blockI*numRows;
217                     end = destIndex + blockRowEnd;
218 
219                     while (destIndex < end) {
220                         dest[destIndex++] = src[srcIndex];
221                         srcIndex += numCols;
222                     }
223                 }
224             }
225         }
226 
227         return dest;
228     }
