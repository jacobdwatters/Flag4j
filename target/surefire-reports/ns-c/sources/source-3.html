66      * Dispatches a matrix transpose problem to the appropriate algorithm based in its shape and size.
67      * @param src Matrix to transpose.
68      * @return The transpose of the source matrix.
69      */
70     public static Matrix dispatch(Matrix src) {
71 
72         double[] dest;
73 
74         Algorithm algorithm = chooseAlgorithm(src.shape);
75 
76         switch(algorithm) {
77             case STANDARD:
78                 dest = RealDenseTranspose.standardMatrix(src.entries, src.numRows, src.numCols);
79                 break;
80             case BLOCKED:
81                 dest = RealDenseTranspose.blockedMatrix(src.entries, src.numRows, src.numCols);
82                 break;
83             case CONCURRENT_STANDARD:
84                 dest = RealDenseTranspose.standardMatrixConcurrent(src.entries, src.numRows, src.numCols);
85                 break;
86             default:
87                 dest = RealDenseTranspose.blockedMatrixConcurrent(src.entries, src.numRows, src.numCols);
88                 break;
89         }
90 
91         return new Matrix(src.numCols, src.numRows, dest);
92     }

213      */
214     private static Algorithm chooseAlgorithmComplex(Shape shape) {
215         Algorithm algorithm;
216 
217         int numEntries = shape.totalEntries().intValueExact();
218 
219         if(numEntries < STANDARD_THRESHOLD) {
220             // Use standard algorithm.
221             algorithm = Algorithm.STANDARD;
222         } else if(numEntries < CONCURRENT_THRESHOLD) {
223             // Use blocked algorithm
224             algorithm = Algorithm.BLOCKED;
225         } else {
226             // Use concurrent blocked implementation.
227             algorithm = Algorithm.CONCURRENT_BLOCKED;
228         }
229 
230         return algorithm;
231     }
