44     /**
45      * Computes the determinant of a square matrix using the LU factorization.
46      * @param A Matrix to compute the determinant of.
47      * @return The determinant of the matrix.
48      * @throws IllegalArgumentException If matrix is not square.
49      */
50     public static double det(Matrix A) {
51         ParameterChecks.assertSquare(A.shape);
52         double det;
53 
54         switch (A.numRows) {
55             case 1:
56                 det = det1(A);
57                 break;
58             case 2:
59                 det = det2(A);
60                 break;
61             case 3:
62                 det = det3(A);
63                 break;
64             default:
65                 det = detLU(A);
66                 break;
67         }
68 
69         return det;
70     }

73     /**
74      * Computes the determinant of a square matrix using the LU factorization.
75      * @param A Matrix to compute the determinant of.
76      * @return The determinant of the matrix.
77      * @throws IllegalArgumentException If matrix is not square.
78      */
79     public static double detLU(Matrix A) {
80         ParameterChecks.assertSquare(A.shape);
81 
82         Matrix[] LU = Decompose.LU(A);
83         Matrix U = LU[2];
84 
85         int numSwaps = (int) (A.numRows - LU[0].trace()) - 1; // Number of swaps in permutation matrix.
86         double detP = Math.pow(-1, numSwaps); // Compute the determinant of P.
87 
88         double detU = 1;
89 
90         // Compute the determinant of U
91         for(int i=0; i<U.numRows; i++) {
92             detU *= U.entries[i*U.numCols + i];
93         }
94 
95         return detP*detU;
96     }

99     /**
100      * Explicitly computes the determinant of a 3x3 matrix.
101      * @param A Matrix to compute the determinant of.
102      * @return The determinant of the 3x3 matrix.
103      */
104     public static double det3(Matrix A) {
105         ParameterChecks.assertEqualShape(A.shape, new Shape(3, 3));
106         double det = A.entries[0]*(A.entries[4]*A.entries[8] - A.entries[5]*A.entries[7]);
107         det -= A.entries[1]*(A.entries[3]*A.entries[8] - A.entries[5]*A.entries[6]);
108         det += A.entries[2]*(A.entries[3]*A.entries[7] - A.entries[4]*A.entries[6]);
109         return det;
110     }
