144     /**
145      * Dynamically chooses matrix multiply algorithm based on the shapes of the two matrices to multiply.
146      * @param shape1 The shape of the first matrix.
147      * @param shape2 The shape fo the second matrix.
148      * @return The algorithm to use in the matrix multiplication.
149      */
150     protected static AlgorithmNames selectAlgorithm(Shape shape1, Shape shape2) {
151         AlgorithmNames name;
152 
153         int rows1 = shape1.get(Axis2D.row());
154         int cols1 = shape1.get(Axis2D.col());
155 
156         // Determine the matrix shape.
157         int matrixShape;
158         if (getRatio(shape1) >= SQUARENESS_RATIO) {
159             matrixShape = 0; // Approximately Square
160         } else if (rows1 > cols1) {
161             matrixShape = 1; // More rows than columns
162         } else {
163             matrixShape = 2; // More columns than rows
164         }
165 
166         // Determine the algorithm name based on the matrix shape and size.
167         switch (matrixShape) {
168             case 0: // Square
169                 if(rows1 < SEQUENTIAL_SWAPPED_THRESHOLD) {
170                     name = AlgorithmNames.REORDERED;
171                 } else if (rows1 < CONCURRENT_SWAPPED_THRESHOLD) {
172                     name = AlgorithmNames.CONCURRENT_REORDERED;
173                 } else {
174                     name = AlgorithmNames.CONCURRENT_BLOCKED_REORDERED;
175                 }
176                 break;
177 
178             case 1: // More rows than columns
179                 if(rows1 <= 100 && cols1 <= 5) {
180                     name = AlgorithmNames.REORDERED;
181                 } else {
182                     name = AlgorithmNames.CONCURRENT_REORDERED;
183                 }
184                 break;
185 
186             case 2: // More columns than rows
187                 if(cols1 <= 100) {
188                     name = (rows1 <= 20) ? AlgorithmNames.REORDERED : AlgorithmNames.CONCURRENT_REORDERED;
189                 } else if(cols1 <= 500) {
190                     name = (rows1 <= 10) ? AlgorithmNames.REORDERED : AlgorithmNames.CONCURRENT_REORDERED;
191                 } else {
192                     if(rows1 <= 5) {
193                         name = AlgorithmNames.REORDERED;
194                     } else if(rows1 <= 50) {
195                         name = AlgorithmNames.CONCURRENT_STANDARD;
196                     } else {
197                         name = AlgorithmNames.CONCURRENT_REORDERED;
198                     }
199                 }
200                 break;
201             default:
202                 name = AlgorithmNames.BLOCKED_REORDERED; // Default to blocked reordered algorithm.
203         }
204 
205         return name;
206     }

234     /**
235      * Computes the squareness ratio of a matrix. This is a value between 0 and 1 with 1 being perfectly
236      * square and 0 being a row/column vector.
237      * @param shape Shape of the matrix to compute the squareness ratio of.
238      * @return The squareness ratio for the specified shape.
239      */
240     private static double getRatio(Shape shape) {
241         int numRows = shape.get(Axis2D.row());
242         int numCols = shape.get(Axis2D.col());
243 
244         double ratio = Math.abs(numRows-numCols);
245         return 1-ratio/Math.max(numRows, numCols);
246     }
