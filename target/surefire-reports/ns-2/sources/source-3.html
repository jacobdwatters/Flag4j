25 package com.flag4j.complex_numbers;

27 /**
28  * A parser for complex numbers represented as a string.
29  */
30 public class CNumberParser {
31 
32     /**
33      * Parses a complex number in the form of a string into its real and imaginary parts.
34      * For example, the string <code>"2+3i"</code> would be parsed into real and imaginary parts
35      * <code>2</code> and <code>3</code> respectively.
36      *
37      * @param num - complex number in one of three forms: <code>a + bi, a,</code> or <code>bi</code> where a and b are
38      * 				real numbers and i is the imaginary unit sqrt(-1)
39      * @return The complex number represented by the string num.
40      */
41     public static CNumber parseNumber(String num) {
42         double[] result = new double[2];
43 
44         CNumberLexer lex = new CNumberLexer(num);
45 
46         CNumberToken token;
47         CNumberToken operator;
48         CNumberToken real;
49         CNumberToken imaginary;
50 
51         token = lex.getNextToken();
52         if(token.matches("im", "i")) { // then we have the imaginary unit
53             result[0] = 0;
54             result[1] = 1;
55         }
56         else {
57             real = token;
58             token = lex.getNextToken();
59 
60             if(token.matches("eof", "")) { // Then we have a real number (a)
61                 result[0] = Double.parseDouble(real.getDetails());
62                 result[1] = 0;
63             }
64             else if(token.matches("im", "i")) { // Then we have a pure imaginary number (bi)
65                 imaginary = real;
66                 result[0] = 0;
67 
68                 token = lex.getNextToken();
69                 token.errorCheck("eof", "");
70 
71                 if(imaginary.getDetails().matches("-")) {
72                     result[1] = -1;
73                 }
74                 else {
75                     result[1] = Double.parseDouble(imaginary.getDetails());
76                 }
77             }
78             else { // Then we have a complex number with nonzero real and imaginary parts (a + bi)
79                 operator = token;
80 
81                 if(!operator.isKind("opp") && operator.isKind("num")) {
82                     imaginary = operator;
83                 }
84                 else {
85                     imaginary = lex.getNextToken();
86                 }
87 
88                 // The operator is negative
89                 if(imaginary.matches("im", "i")) { // Then we have the unit imaginary number
90                     token = lex.getNextToken();
91                     token.errorCheck("eof", "");
92 
93                     result[0] = Double.parseDouble(real.getDetails());
94                     result[1] = 1;
95 
96                 }
97                 else { // Then we have a multiple of the unit imaginary number
98                     imaginary.errorCheck("num");
99 
100                     token = lex.getNextToken();
101                     token.errorCheck("im", "i");
102 
103                     token = lex.getNextToken();
104                     token.errorCheck("eof", "");
105 
106                     result[0] = Double.parseDouble(real.getDetails());
107                     result[1] = Double.parseDouble(imaginary.getDetails());
108 
109                 }
110                 if(operator.getDetails().equals("-")) { // The operator is negative
111                     result[1] = -result[1];
112                 }
113             }
114         }
115 
116         return new CNumber(result[0], result[1]);
117     }
