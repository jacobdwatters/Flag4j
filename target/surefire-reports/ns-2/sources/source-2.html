102     /**
103      * Produces next {@link CNumberToken} from complex number string. Also removes this
104      * CNumberToken from the string. This method implements a finite automata which describes the legal arrangement of
105      * tokens within a complex number.
106      *
107      * @return Next {@link CNumberToken} in string.
108      * @throws RuntimeException If the string is not a valid representation of a complex number.
109      */
110     public CNumberToken getNextToken() {
111         int state = 1;  // State of Finite Automata
112         boolean done = false;
113         StringBuilder dataBuilder = new StringBuilder(); // specific info for the CNumberToken
114         int sym;  // holds current symbol
115 
116 
117         do {
118             sym = getNextSymbol(); // Will return -1 if there is no symbol to get
119 
120             if(state == 1) {
121                 if(sym == 45) {
122                     state = 2;
123                     dataBuilder.append((char) sym);
124                 }
125                 else if(sym == 43) {
126                     state = 4;
127                     dataBuilder.append((char) sym);
128                     done = true;
129                 }
130                 else if(digit(sym)) {
131                     state = 3;
132                     dataBuilder.append((char) sym);
133                 }
134                 else if(sym == 'i' || sym == 'j') {
135                     state = 6;
136                     dataBuilder.append((char) sym);
137                     done = true;
138                 }
139                 else if(sym == -1) { // We have reached the end of the string
140                     state = 5;
141                     done = true;
142                 }
143                 else if(!(sym == 9 || sym == 10 || sym == 13 ||
144                         sym == 32)){
145                     // Otherwise, if it is not a whitespace character, we have encountered an unexpected symbol.
146                     error(String.format(ERR_MSG, (char) sym));
147                 }
148             }
149 
150             else if(state == 2) {
151                 if(sym == 9 || sym == 10 || sym == 13 ||
152                         sym == 32) { // Whitespace
153                     putBackSymbol(sym);
154                     done = true;
155                 }
156                 else if(digit(sym)) {
157                     state = 3;
158                     dataBuilder.append((char) sym);
159                 }
160                 else {
161                     putBackSymbol(sym);
162                     done = true;
163                 }
164             }
165             else if(state == 3) {
166                 if(digit(sym)) {
167                     // State does not need to change here.
168                     dataBuilder.append((char) sym);
169                 }
170                 else if(sym == '.') {
171                     state = 7;
172                     dataBuilder.append((char) sym);
173                 }
174                 else {
175                     putBackSymbol(sym);
176                     done = true;
177                 }
178             }
179             else if(state == 7) {
180                 if(digit(sym)) {
181                     state = 8;
182                     dataBuilder.append((char) sym);
183                 }
184                 else {
185                     error(String.format(ERR_MSG, (char) sym));
186                 }
187             }
188             else if(state == 8) {
189                 if(digit(sym)) {
190                     // State does not need to change here.
191                     dataBuilder.append((char) sym);
192                 }
193                 else {
194                     putBackSymbol(sym);
195                     done = true;
196                 }
197             }
198 
199         } while(!done);
200 
201         if(state == 2 || state == 4) { // we have an operator
202             return new CNumberToken("opp", dataBuilder.toString());
203         }
204         else if(state == 3 || state == 8) { // we have a number
205             return new CNumberToken("num", dataBuilder.toString());
206         }
207         else if(state == 5) { // end of number
208             return new CNumberToken("eof", dataBuilder.toString());
209         }
210         else if(state == 6) { // we have the imaginary unit
211             return new CNumberToken("im","i");
212         }
213         else {
214             error("somehow Lexer FA halted in bad state " + state);
215             return null;
216         }
217     }
