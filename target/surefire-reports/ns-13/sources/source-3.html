48      * Computes the matrix multiplication between a real dense matrix and a real sparse matrix using a standard algorithm.
49      * @param src1 Entries of the dense matrix.
50      * @param shape1 Shape of the dense matrix.
51      * @param src2 Non-zero entries of the sparse matrix.
52      * @param rowIndices Row indices for non-zero entries of the sparse matrix.
53      * @param colIndices Column indices for non-zero entries of the sparse matrix.
54      * @param shape2 Shape of the sparse matrix.
55      * @return The result of the matrix multiplication.
56      */
57     public static double[] standard(double[] src1, Shape shape1, double[] src2,
58                                     int[] rowIndices, int[] colIndices, Shape shape2) {
59         int rows1 = shape1.dims[Axis2D.row()];
60         int cols1 = shape1.dims[Axis2D.col()];
61         int cols2 = shape2.dims[Axis2D.col()];
62 
63         double[] dest = new double[rows1*cols2];
64 
65         int row, col;
66         int destStart, src1Start;
67 
68         for(int i=0; i<rows1; i++) {
69             destStart = i*cols2;
70             src1Start = i*cols1;
71 
72             // Loop over non-zero entries of sparse matrix.
73             for(int j=0; j<src2.length; j++) {
74                 row = rowIndices[j];
75                 col = colIndices[j];
76 
77                 dest[destStart + col] += src1[src1Start + row]*src2[j];
78             }
79         }
80 
81         return dest;
82     }

193         int denseRows = shape1.dims[Axis2D.row()];
194         int denseCols = shape1.dims[Axis2D.col()];
195         int nonZeros = src2.length;
196 
197         double[] dest = new double[denseRows];
198         int k;
199 
200         for(int i=0; i<denseRows; i++) {
201             for(int j=0; j<nonZeros; j++) {
202                 k = indices[j];
203                 dest[i] += src1[i*denseCols + k]*src2[j];
204             }
205         }

239      * Computes the dense matrix sparse vector multiplication using a blocked algorithm.
240      * @param src1 Entries of the dense matrix.
241      * @param shape1 Shape of the dense matrix.
242      * @param src2 Non-zero entries of the sparse vector.
243      * @param indices Indices of non-zero entries in sparse vector.
244      * @return Entries of the dense matrix resulting from the matrix vector multiplication.
245      */
246     public static double[] blockedVector(double[] src1, Shape shape1, double[] src2, int[] indices) {
247         int rows1 = shape1.dims[Axis2D.row()];
248         int cols1 = shape1.dims[Axis2D.col()];
249         int rows2 = src2.length;
250 
251         int bsize = Configurations.getBlockSize(); // Get the block size to use.
252 
253         double[] dest = new double[rows1];
254         int k;
255 
256         // Blocked matrix-vector multiply
257         for(int ii=0; ii<rows1; ii += bsize) {
258             for(int jj=0; jj<rows2; jj += bsize) {
259                 // Multiply the current blocks
260                 for(int i=ii; i<ii+bsize && i<rows1; i++) {
261                     for(int j=jj; j<jj+bsize && j<rows2; j++) {
262                         k = indices[j];
263                         dest[i] += src1[i*cols1 + k]*src2[j];
264                     }
265                 }
266             }
267         }
268 
269         return dest;
270     }
