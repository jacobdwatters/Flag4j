34 /**
35  * This class contains methods for checking the equality of real sparse tensors.
36  */
37 public class RealSparseEquals {
38 
39 
40     /**
41      * Checks if two real sparse tensors are real. Assumes the indices of each sparse tensor are sorted.
42      * @param a First tensor in the equality check.
43      * @param b Second tensor in the equality check.
44      * @return True if the tensors are equal. False otherwise.
45      */
46     public static boolean tensorEquals(SparseTensor a, SparseTensor b) {
47         return a.shape.equals(b.shape) && Arrays.equals(a.entries, b.entries)
48                 && Arrays.deepEquals(a.indices, b.indices);
49     }
50 
51 
52     /**
53      * Checks if two real sparse matrices are real. Assumes the indices of each sparse matrix are sorted.
54      * @param a First matrix in the equality check.
55      * @param b Second matrix in the equality check.
56      * @return True if the matrices are equal. False otherwise.
57      */
58     public static boolean matrixEquals(SparseMatrix a, SparseMatrix b) {
59         return a.shape.equals(b.shape) && Arrays.equals(a.entries, b.entries)
60                 && Arrays.equals(a.rowIndices, b.rowIndices) && Arrays.equals(a.colIndices, b.colIndices);
61     }
62 
63 
64     /**
65      * Checks if two real sparse vectors are real. Assumes the indices of each sparse vector are sorted.
66      * @param a First vector in the equality check.
67      * @param b Second vector in the equality check.
68      * @return True if the vectors are equal. False otherwise.
69      */
70     public static boolean vectorEquals(SparseVector a, SparseVector b) {
71         return a.size == b.size && Arrays.equals(a.indices, b.indices)
72                 && Arrays.equals(a.entries, b.entries);
73     }
