51     /**
52      * Centers a string within a specified size bin.
53      * @param s String to center.
54      * @param size Size of the bin to center string within.
55      * @param pad Padding character.
56      * @return A string made up of the padding character of specified size with the string s centered within it.
57      */
58     public static String center(String s, int size, String pad) {
59         if (s == null || size <= s.length())
60             return s;
61 
62         StringBuilder sb = new StringBuilder(size);
63         sb.append(String.valueOf(pad).repeat((size - s.length()) / 2));
64         sb.append(s);
65         while (sb.length() < size) {
66             sb.append(pad);
67         }
68 
69         return sb.toString();
70     }
