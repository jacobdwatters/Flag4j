70     /**
71      * Generates a uniformly distributed pseudorandom complex number with given magnitude.
72      * @param mag Magnitude of the complex number. Must be non-negative.
73      * @return random complex number with specified magnitude.
74      * @throws IllegalArgumentException If the magnitude is negative.
75      */
76     public CNumber random(double mag) {
77         if(mag < 0) {
78             throw new IllegalArgumentException("Magnitude must be non-negative but got " + mag);
79         }
80 
81         // Generate real component.
82         double real = rng.nextDouble()*mag;
83 
84         // Compute imaginary component using Pythagorean theorem.
85         double imaginary = Math.sqrt(Math.pow(mag, 2) - Math.pow(real, 2));
86 
87         // Choose complex plane quadrant (i.e. signs of each component).
88         real = rng.nextBoolean() ? real : -real;
89         imaginary = rng.nextBoolean() ? imaginary : -imaginary;
90 
91         return new CNumber(real, imaginary);
92     }

95     /**
96      * Generates a pseudorandom complex number with magnitude uniformly distributed in {@code [min, max)}.
97      *
98      * @param min Minimum value for random number. Must be non-negative.
99      * @param max Maximum value for random number. Must be larger than or equal to min.
100      * @return A pseudorandom complex number with magnitude uniformly distributed in {@code [min, max)}.
101      * @throws IllegalArgumentException If {@code min} is not positive or if {@code max} is less than {@code min}.
102      */
103     public CNumber random(double min, double max) {
104         if(min < 0) {
105             throw new IllegalArgumentException("Min value must be non-negative but got " + min + ".");
106         }
107 
108         if(min > max) {
109             throw new IllegalArgumentException("Max value must be greater than or equal to min but got min=" +
110                     min + " and max=" + max + ".");
111         }
112 
113         return random(rng.nextDouble()*(max-min) + min);
114     }
