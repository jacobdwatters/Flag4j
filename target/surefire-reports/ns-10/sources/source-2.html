46     /**
47      * Computes the determinant of a square matrix using the LU factorization.
48      * @param A Matrix to compute the determinant of.
49      * @return The determinant of the matrix.
50      * @throws IllegalArgumentException If matrix is not square.
51      */
52     public static double det(Matrix A) {
53         ParameterChecks.assertSquare(A.shape);
54         double det;
55 
56         switch (A.numRows) {
57             case 1:
58                 det = det1(A);
59                 break;
60             case 2:
61                 det = det2(A);
62                 break;
63             case 3:
64                 det = det3(A);
65                 break;
66             default:
67                 det = detLU(A);
68                 break;
69         }
70 
71         return det;
72     }

81     public static double detLU(Matrix A) {
82         ParameterChecks.assertSquare(A.shape);
83         Matrix[] LU = Decompose.lu(A);

85         return detLU(LU[0], LU[1], LU[2]);

89     /**
90      * Computes the determinant for a matrix which has been factored into a unit lower triangular matrix {@code L}
91      * and an upper triangular matrix {@code U} using partial pivoting (i.e. row swaps).
92      * @param P Row permutation matrix in the {@code LU} decomposition.
93      * @param L Unit lower triangular matrix.
94      * @param U Upper triangular matrix.
95      * @return The determinant of the matrix which has been factored into a unit lower triangular matrix {@code L}
96      * and an upper triangular matrix {@code U} using partial pivoting.
97      */
98     public static double detLU(Matrix P, Matrix L, Matrix U) {
99         int numSwaps = (int) (L.numRows - P.trace()) - 1; // Number of swaps in permutation matrix.
100         double detP = Math.pow(-1, numSwaps); // Compute the determinant of P.
101         return detP*detLU(L, U);
102     }

105     /**
106      * Computes the determinant for a matrix which has been factored into a unit lower triangular matrix {@code L}
107      * and an upper triangular matrix {@code U} with no pivoting.
108      * @param L Unit lower triangular matrix.
109      * @param U Upper triangular matrix.
110      * @return The determinant of the matrix which has been factored into a unit lower triangular matrix {@code L}
111      * and an upper triangular matrix {@code U}.
112      * @see com.flag4j.linalg.decompositions.LUDecomposition
113      * @see com.flag4j.linalg.decompositions.RealLUDecomposition
114      */
115     public static double detLU(Matrix L, Matrix U) {
116         double detU = 1;
117 
118         // Compute the determinant of U
119         for(int i=0; i<U.numRows; i++) {
120             detU *= U.entries[i*U.numCols + i];
121         }
122 
123         return detU;
124     }

128      * Computes the determinant for a matrix which has been factored into a unit lower triangular matrix {@code L}
129      * and an upper triangular matrix {@code U} with no pivoting.
130      * @param L Unit lower triangular matrix.
131      * @param U Upper triangular matrix.
132      * @return The determinant of the matrix which has been factored into a unit lower triangular matrix {@code L}
133      * and an upper triangular matrix {@code U}.

135      * @see com.flag4j.linalg.decompositions.ComplexLUDecomposition
136      */
137     public static CNumber detLU(CMatrix L, CMatrix U) {
138         CNumber detU = new CNumber(1);
139 
140         // Compute the determinant of U
141         for(int i=0; i<U.numRows; i++) {
142             detU.multEq(U.entries[i*U.numCols + i]);
143         }
144 
145         return detU;
146     }

149     /**
150      * Explicitly computes the determinant of a 3x3 matrix.
151      * @param A Matrix to compute the determinant of.
152      * @return The determinant of the 3x3 matrix.
153      */
154     public static double det3(Matrix A) {
155         ParameterChecks.assertEqualShape(A.shape, new Shape(3, 3));
156         double det = A.entries[0]*(A.entries[4]*A.entries[8] - A.entries[5]*A.entries[7]);
157         det -= A.entries[1]*(A.entries[3]*A.entries[8] - A.entries[5]*A.entries[6]);
158         det += A.entries[2]*(A.entries[3]*A.entries[7] - A.entries[4]*A.entries[6]);
159         return det;
160     }
