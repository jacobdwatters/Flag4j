47      * Transposes tensor along specified axes using a standard transpose algorithm. In this context, transposing a
48      * tensor is equivalent to swapping a pair of axes.
49      * @param src Entries of the tensor.
50      * @param shape Shape of the tensor to transpose.
51      * @param axis1 First axis to swap in transpose.
52      * @param axis2 Second axis to swap in transpose.
53      * @return The transpose of the tensor along the specified axes.
54      */
55     public static double[] standard(final double[] src, final Shape shape, final int axis1, final int axis2) {
56         if(shape.getRank() < 2) { // Can't transpose tensor with less than 2 axes.
57             throw new IllegalArgumentException("Tensor transpose not defined for rank " + shape.getRank() +
58                     " tensor.");
59         }
60 
61         double[] dest = new double[shape.totalEntries().intValue()];
62         Shape destShape = shape.copy().swapAxes(axis1, axis2);
63         int[] destIndices;
64 
65         for(int i=0; i<src.length; i++) {
66             destIndices = shape.getIndices(i);
67             ArrayUtils.swap(destIndices, axis1, axis2); // Compute destination indices.
68             dest[destShape.entriesIndex(destIndices)] = src[i]; // Apply transpose for the element
69         }
70 
71         return dest;
72     }

170      * Transposes a matrix using the standard algorithm.
171      * @param src Entries of the matrix to transpose.
172      * @param numRows Number of rows in the matrix.
173      * @param numCols Number of columns in the matrix.
174      * @return The transpose of the matrix.
175      */
176     public static double[] standardMatrix(final double[] src, final int numRows, final int numCols) {
177         double[] dest = new double[numRows*numCols];
178 
179         int destIndex, srcIndex, end;
180 
181         for (int i=0; i<numCols; i++) {
182             srcIndex = i;
183             destIndex = i*numRows;
184             end = destIndex + numRows;
185 
186             while (destIndex < end) {
187                 dest[destIndex++] = src[srcIndex];
188                 srcIndex += numCols;
189             }
190         }
191 
192         return dest;
193     }

197      * Transposes a matrix using a blocked algorithm. To get or set the block size see
198      * {@link Configurations#getBlockSize()} or {@link Configurations#setBlockSize(int)}.
199      * @param src Source entries to compute transpose of.
200      * @param numRows Number of rows in the matrix.
201      * @param numCols Number of columns in the matrix.
202      * @return The transpose of this tensor along specified axes
203      */
204     public static double[] blockedMatrix(final double[] src, final int numRows, final int numCols) {
205         double[] dest = new double[numRows*numCols];
206         final int blockSize = Configurations.getBlockSize();
207         int srcIndexStart, destIndexStart;
208         int srcIndex, destIndex, srcIndexEnd, destIndexEnd;
209         int blockHeight;
210 
211         for(int ii=0; ii<numRows; ii+=blockSize) {
212             blockHeight = Math.min(ii+blockSize, numRows) - ii;
213             srcIndexStart = ii*numCols;
214             destIndexStart = ii;
215 
216             for(int jj=0; jj<numCols; jj+=blockSize) {
217                 srcIndexEnd = srcIndexStart + Math.min(numCols-jj, blockSize);
218 
219                 while(srcIndexStart<srcIndexEnd) {
220                     srcIndex = srcIndexStart;
221                     destIndex = destIndexStart;
222                     destIndexEnd = destIndex + blockHeight;
223 
224                     while(destIndex<destIndexEnd) {
225                         dest[destIndex++] = src[srcIndex];
226                         srcIndex+=numCols;
227                     }
228 
229                     destIndexStart += numRows;
230                     srcIndexStart++;
231                 }
232             }
233         }
234 
235         return dest;
236     }
