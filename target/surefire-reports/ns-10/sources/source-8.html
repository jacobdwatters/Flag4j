214      * Compute the L<sub>p, q</sub> norm of a matrix.
215      * @param src Entries of the matrix.
216      * @param shape Shape of the matrix.
217      * @param p First parameter in L<sub>p, q</sub> norm.
218      * @param q Second parameter in L<sub>p, q</sub> norm.
219      * @return The L<sub>p, q</sub> norm of the matrix.
220      * @throws IllegalArgumentException If {@code p} or {@code q} is less than 1.
221      */
222     public static double matrixNormLpq(double[] src, Shape shape, double p, double q) {
223         ParameterChecks.assertGreaterEq(1, p, q);
224 
225         double norm = 0;
226         double colSum;
227         int rows = shape.dims[Axis2D.row()];
228         int cols = shape.dims[Axis2D.col()];
229 
230         for(int j=0; j<cols; j++) {
231             colSum=0;
232             for(int i=0; i<rows; i++) {
233                 colSum += Math.pow(Math.abs(src[i*cols + j]), p);
234             }
235             norm += Math.pow(colSum, q/p);
236         }
237 
238         return Math.pow(norm, 1/q);
239     }
