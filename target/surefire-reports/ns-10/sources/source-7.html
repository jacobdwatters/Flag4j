46      * Computes the matrix multiplication between two real dense matrices using the standard algorithm.
47      * @param src1 Entries of the first matrix.
48      * @param shape1 Shape of the first matrix.
49      * @param src2 Entries of the second matrix.
50      * @param shape2 Shape fo the second matrix.
51      * @return The result of matrix multiplying the two matrices.

53     public static double[] standard(double[] src1, Shape shape1, double[] src2, Shape shape2) {
54         int rows1 = shape1.dims[Axis2D.row()];
55         int rows2 = shape2.dims[Axis2D.row()];
56         int cols2 = shape2.dims[Axis2D.col()];
57 
58         double[] dest = new double[rows1*cols2];
59         int src1Index, src2Index, destIndex, src1IndexStart, destIndexStart, end;
60 
61         for(int i=0; i<rows1; i++) {
62             src1IndexStart = i*rows2;
63             destIndexStart = i*cols2;
64 
65             for(int j=0; j<cols2; j++) {
66                 src2Index = j;
67                 src1Index = src1IndexStart;
68                 destIndex = destIndexStart + j;
69                 end = src1Index + rows2;
70 
71                 while(src1Index<end) {
72                     dest[destIndex] += src1[src1Index++]*src2[src2Index];
73                     src2Index += cols2;
74                 }
75             }
76         }
77 
78         return dest;
79     }

130         int rows1 = shape1.dims[Axis2D.row()];

132         int cols1 = shape1.dims[Axis2D.col()];
133 
134         double[] dest = new double[rows1 * cols2];
135         int blockSize = Configurations.getBlockSize();
136         int iBound, jBound, kBound;
137         int src1Start, destStart, stopIndex;
138         int destIndex, src1Index, src2Index;
139 
140         for(int ii=0; ii<rows1; ii+=blockSize) {
141             iBound = Math.min(ii + blockSize, rows1);
142             for(int jj = 0; jj<cols2; jj+=blockSize) {
143                 jBound = Math.min(jj + blockSize, cols2);
144                 for(int kk = 0; kk<cols1; kk+=blockSize) {
145                     kBound = Math.min(kk + blockSize, cols1);
146 
147                     // Multiply current blocks.

149                         src1Start = i*cols1 + kk;
150                         stopIndex = src1Start+(kBound-kk);

152

154                             destIndex = destStart + j;
