49      * Computes the vector inner product between a complex dense vector and a complex sparse vector.
50      * @param src1 Entries of the dense vector.
51      * @param src2 Non-zero entries of the sparse vector.
52      * @param indices Indices of nonzero values in sparse vector.
53      * @param sparseSize Full size of the sparse vector (i.e. total number of entries including zeros).
54      * @return The inner product of the two vectors.
55      * @throws IllegalArgumentException If the number of entries in the two vectors is not equivalent.
56      */
57     public static CNumber innerProduct(CNumber[] src1, CNumber[] src2, int[] indices, int sparseSize) {
58         ParameterChecks.assertArrayLengthsEq(src1.length, sparseSize);
59         CNumber innerProd = new CNumber();
60         int index;
61 
62         for(int i=0; i<src2.length; i++) {
63             index = indices[i];
64             innerProd.addEq(src2[i].conj().mult(src1[index]));
65         }
66

96      * @param src1 Entries of the dense vector.
97      * @param src2 Non-zero entries of the sparse vector.
98      * @param indices Indices of non-zero entries of sparse vector.

101     public static CNumber[] outerProduct(CNumber[] src1, CNumber[] src2, int[] indices, int sparseSize) {
102         CNumber[] dest = new CNumber[src1.length*sparseSize];
103         ArrayUtils.fillZeros(dest);
104         int index;
105 
106         for(int i=0; i<src1.length; i++) {
107             for(int j=0; j<src2.length; j++) {
108                 index = indices[j];
109                 dest[i*sparseSize + index] = src1[i].mult(src2[j].conj());
110             }
111         }
112 
113         return dest;
114     }
