50      * Computes the matrix multiplication between a complex dense matrix and a complex sparse matrix using a standard algorithm.
51      * @param src1 Entries of the dense matrix.
52      * @param shape1 Shape of the dense matrix.
53      * @param src2 Non-zero entries of the sparse matrix.
54      * @param rowIndices Row indices for non-zero entries of the sparse matrix.
55      * @param colIndices Column indices for non-zero entries of the sparse matrix.
56      * @param shape2 Shape of the sparse matrix.
57      * @return The result of the matrix multiplication.
58      */
59     public static CNumber[] standard(CNumber[] src1, Shape shape1, CNumber[] src2,
60                                     int[] rowIndices, int[] colIndices, Shape shape2) {
61         int rows1 = shape1.dims[Axis2D.row()];
62         int cols1 = shape1.dims[Axis2D.col()];
63         int cols2 = shape2.dims[Axis2D.col()];
64 
65         CNumber[] dest = new CNumber[rows1*cols2];
66         ArrayUtils.fillZeros(dest);
67 
68         int row, col;
69 
70         for(int i=0; i<rows1; i++) {
71             // Loop over non-zero entries of sparse matrix.
72             for(int j=0; j<src2.length; j++) {
73                 row = rowIndices[j];
74                 col = colIndices[j];
75 
76                 dest[i*cols2 + col].addEq(src1[i*cols1 + row].mult(src2[j]));
77             }
78         }
79 
80         return dest;
81     }

245      * Computes the dense matrix sparse vector multiplication using a blocked algorithm.
246      * @param src1 Entries of the dense matrix.
247      * @param shape1 Shape of the dense matrix.
248      * @param src2 Non-zero entries of the sparse vector.
249      * @param indices Indices of non-zero entries in sparse vector.
250      * @return Entries of the dense matrix resulting from the matrix vector multiplication.
251      */
252     public static CNumber[] blockedVector(CNumber[] src1, Shape shape1, CNumber[] src2, int[] indices) {
253         int rows1 = shape1.dims[Axis2D.row()];
254         int cols1 = shape1.dims[Axis2D.col()];
255         int rows2 = src2.length;
256 
257         int bsize = Configurations.getBlockSize(); // Get the block size to use.
258 
259         CNumber[] dest = new CNumber[rows1];
260         ArrayUtils.fillZeros(dest);
261         int k;
262 
263         // Blocked matrix-vector multiply
264         for(int ii=0; ii<rows1; ii += bsize) {
265             for(int jj=0; jj<rows2; jj += bsize) {
266                 // Multiply the current blocks
267                 for(int i=ii; i<ii+bsize && i<rows1; i++) {
268                     for(int j=jj; j<jj+bsize && j<rows2; j++) {
269                         k = indices[j];
270                         dest[i].addEq(src1[i*cols1 + k].mult(src2[j]));
271                     }
272                 }
273             }
274         }
275 
276         return dest;
277     }
