46     /**
47      * Computes the tensor dot product along the first tensors last axis and the second tensors second-to-last axis.
48      * @param src1 First tensor in the tensor product.
49      * @param src2 Second tensor in the tensor product.
50      * @return The tensor dot product along the first tensors last axis and the second tensors second-to-last axis.
51      * @throws IllegalArgumentException If this tensors shape along the last axis does not match {@code src2} shape
52      * along the second-to-last axis.
53      */
54     public static CTensor dot(CTensor src1, CTensor src2) {
55         int src1Dim = src1.getRank(); // Rank of first tensor.
56         int src2Dim = src2.getRank(); // Rank of second tensor.
57 
58         // Ensure tensors have same length along last axis.
59         ParameterChecks.assertEquals(src1.shape.get(src1Dim-1), src2.shape.get(src2Dim-2));
60 
61         int axisLength = src1.shape.get(src1Dim-1); // Length of axis along which to compute dot product.
62         CTensor src2Swap = src2.T(src2Dim-1, src2Dim-2);
63 
64         int iStop = src1.totalEntries().intValueExact()/src1.shape.get(src1Dim-1);
65         int jStop = src2.totalEntries().intValueExact()/src2.shape.get(src2Dim-2);
66         int[] dims = new int[src1Dim+src2Dim-2];
67         int idx = 0;
68 
69         // Copy shape dimensions from each tensor.
70         for(int i=0; i<src1Dim-1; i++) {
71             dims[idx++] = src1.shape.dims[i];
72         }
73 
74         for(int i=0; i<src2Dim-1; i++) {
75             dims[idx++] = src2Swap.shape.dims[i];
76         }
77 
78         Shape destShape = new Shape(dims);
79         CNumber[] dest = new CNumber[destShape.totalEntries().intValueExact()];
80         int src1_start, src2_start;
81 
82         idx = 0;
83         for(int i=0; i<iStop; i++) {
84             src1_start = i*axisLength;
85 
86             for(int j=0; j<jStop; j++) {
87                 src2_start = j*axisLength;
88 
89                 for(int k=0; k<axisLength; k++) {
90                     dest[idx].addEq(src1.entries[src1_start + k].mult(src2Swap.entries[src2_start + k]));
91                 }
92 
93                 idx++;
94             }
95         }
96 
97         return new CTensor(destShape, dest);
98     }

101     /**
102      * Computes the tensor contraction of this tensor with a specified tensor over the specified set of axes. That is,
103      * computes the sum of products between the two tensors along the specified set of axes.
104      * @param src1 First tensor in the contraction.
105      * @param src2 Second tensor in the contraction.
106      * @param src1Axes Axes along which to compute products for {@code src1} tensor.
107      * @param src2Axes Axes along which to compute products for {@code src2} tensor.
108      * @return The tensor dot product over the specified axes.
109      * @throws IllegalArgumentException If the two tensors shapes do not match along the specified axes pairwise in
110      * {@code aAxes} and {@code bAxes}.
111      * @throws IllegalArgumentException If {@code aAxes} and {@code bAxes} do not match in length, or if any of the axes
112      * are out of bounds for the corresponding tensor.
113      */
114     public static CTensor tensorDot(CTensor src1, CTensor src2, int[] src1Axes, int[] src2Axes) {
115         // Each array must specify the same number of axes.
116         ParameterChecks.assertEquals(src1Axes.length, src2Axes.length);
117 
118         // Axis values must be less than the rank of the tensor and non-negative
119         ParameterChecks.assertLessEq(src1.getRank()-1, src1Axes);
120         ParameterChecks.assertGreaterEq(0, src1Axes);
121         ParameterChecks.assertLessEq(src2.getRank()-1, src2Axes);
122         ParameterChecks.assertGreaterEq(0, src2Axes);
123 
124         int[] notin;
125         int n1;
126         int n2;
127         int pos;
128 
129         // ---- Compute new axes and shapes for first tensor. ----
130         notin = ArrayUtils.notinAxes(src1Axes, src1.getRank());
131         int[] src1NewAxes = ArrayUtils.join(notin, src1Axes);
132 
133         n2 = 1;
134         for(int axis : src1Axes) {
135             n2 *= src1.shape.get(axis);
136         }
137 
138         n1 = 1;
139         int[] src1OldDims = new int[notin.length];
140         pos = 0;
141         for(int axis : notin) {
142             n1 *= src1.shape.get(axis);
143             src1OldDims[pos++] = src1.shape.get(axis);
144         }
145 
146         Shape src1NewShape = new Shape(n1, n2);
147         // -----------------------------------------------------
148 
149         // ---- Compute new axes and shapes for second tensor. ----
150         notin = ArrayUtils.notinAxes(src2Axes, src2.getRank());
151         int[] src2NewAxes = ArrayUtils.join(src2Axes, notin);
152 
153         n2 = 1;
154         for(int axis : src2Axes) {
155             n2 *= src2.shape.get(axis);
156         }
157 
158         n1 = 1;
159         pos = 0;
160         int[] src2OldDims = new int[notin.length];
161         for(int axis : notin) {
162             n1 *= src2.shape.get(axis);
163             src2OldDims[pos++] = src2.shape.get(axis);
164         }
165 
166         Shape src2NewShape = new Shape(n2, n1);
167         // -----------------------------------------------------
168 
169         // Reform problem as a matrix multiplication problem.
170         CMatrix at = new CMatrix(
171                 src1NewShape,
172                 ComplexDenseTranspose.standardConcurrent(src1.entries, src1.shape, src1NewAxes)
173         );
174         CMatrix bt = new CMatrix(
175                 src2NewShape,
176                 ComplexDenseTranspose.standardConcurrent(src2.entries, src2.shape, src2NewAxes)
177         );
178 
179         CNumber[] destEntries = MatrixMultiplyDispatcher.dispatch(at, bt);
180         Shape destShape = new Shape(ArrayUtils.join(src1OldDims, src2OldDims));
181 
182         return new CTensor(destShape, destEntries);
183     }
