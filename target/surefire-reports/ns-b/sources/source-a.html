47      * Transposes tensor along specified axes using a standard transpose algorithm. In this context, transposing a
48      * tensor is equivalent to swapping a pair of axes.
49      * @param src Entries of the tensor.
50      * @param shape Shape of the tensor to transpose.
51      * @param axis1 First axis to swap in transpose.
52      * @param axis2 Second axis to swap in transpose.
53      * @return The transpose of the tensor along the specified axes.
54      */
55     public static CNumber[] standard(final CNumber[] src, final Shape shape, final int axis1, final int axis2) {
56         if(shape.getRank() < 2) { // Can't transpose tensor with less than 2 axes.
57             throw new IllegalArgumentException("Tensor transpose not defined for rank " + shape.getRank() +
58                     " tensor.");
59         }
60 
61         CNumber[] dest = new CNumber[shape.totalEntries().intValue()];
62         Shape destShape = shape.copy().swapAxes(axis1, axis2);
63         int[] destIndices;
64 
65         for(int i=0; i<src.length; i++) {
66             destIndices = shape.getIndices(i);
67             ArrayUtils.swap(destIndices, axis1, axis2); // Compute destination indices.
68             dest[destShape.entriesIndex(destIndices)] = src[i].copy(); // Apply transpose for the element
69         }
70 
71         return dest;
72     }

167      * Transposes a matrix using the standard algorithm.
168      * @param src Entries of the matrix to transpose.
169      * @param numRows Number of rows in the matrix.
170      * @param numCols Number of columns in the matrix.
171      * @return The transpose of the matrix.

173     public static CNumber[] standardMatrix(final CNumber[] src, final int numRows, final int numCols) {
174         CNumber[] dest = new CNumber[numRows*numCols];
175 
176         int destIndex, srcIndex, end;
177 
178         for (int i=0; i<numCols; i++) {
179             srcIndex = i;
180             destIndex = i*numRows;
181             end = destIndex + numRows;
182 
183             while (destIndex < end) {
184                 dest[destIndex++] = src[srcIndex].copy();
185                 srcIndex += numCols;
186             }
187         }
188 
189         return dest;

196      * @param src Source matrix in the transpose.

201     public static CNumber[] blockedMatrix(final CNumber[] src, final int numRows, final int numCols) {
202         CNumber[] dest = new CNumber[numRows*numCols];
203         final int blockSize = Configurations.getBlockSize();
204         int blockRowEnd;
205         int blockColEnd;
206         int srcIndex, destIndex, end;
207 
208         for(int i=0; i<numCols; i+=blockSize) {
209             for(int j=0; j<numRows; j+=blockSize) {
210                 blockRowEnd = Math.min(blockSize, numRows);
211                 blockColEnd = Math.min(i+blockSize, numCols);
212 
213                 // Transpose the block beginning at (i, j)
214                 for(int blockI=i; blockI<blockColEnd; blockI++) {
215                     srcIndex = blockI;
216                     destIndex = blockI*numRows;
217                     end = destIndex + blockRowEnd;
218 
219                     while (destIndex < end) {
220                         dest[destIndex++] = src[srcIndex].copy();
221                         srcIndex += numCols;
222                     }
223                 }
