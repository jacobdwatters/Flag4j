57     public static CNumber[] standard(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
58         int rows1 = shape1.dims[Axis2D.row()];
59         int rows2 = shape2.dims[Axis2D.row()];
60         int cols2 = shape2.dims[Axis2D.col()];
61 
62         CNumber[] dest = new CNumber[rows1*cols2];
63         ArrayUtils.fill(dest, CNumber.ZERO);
64 
65         int src1Index, src2Index, destIndex, src1IndexStart, destIndexStart, end;
66 
67         for(int i=0; i<rows1; i++) {
68             src1IndexStart = i*rows2;
69             destIndexStart = i*cols2;
70 
71             for(int j=0; j<cols2; j++) {
72                 src2Index = j;
73                 src1Index = src1IndexStart;
74                 destIndex = destIndexStart + j;
75                 end = src1Index + rows2;
76 
77                 while(src1Index<end) {
78                     dest[destIndex].addEq(src1[src1Index++].mult(src2[src2Index]));
79                     src2Index += cols2;
80                 }
81             }
82         }
83 
84         return dest;
85     }

136     public static CNumber[] blocked(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
137         int rows1 = shape1.dims[Axis2D.row()];
138         int cols2 = shape2.dims[Axis2D.col()];
139         int cols1 = shape1.dims[Axis2D.col()];
140 
141         CNumber[] dest = new CNumber[rows1 * cols2];
142         ArrayUtils.fill(dest, CNumber.ZERO);
143 
144         int blockSize = Configurations.getBlockSize();
145         int iBound, jBound, kBound;
146         int src1Start, destStart, stopIndex;
147         int destIndex, src1Index, src2Index;
148 
149         for(int ii=0; ii<rows1; ii+=blockSize) {
150             iBound = Math.min(ii + blockSize, rows1);
151             for(int jj = 0; jj<cols2; jj+=blockSize) {
152                 jBound = Math.min(jj + blockSize, cols2);
153                 for(int kk = 0; kk<cols1; kk+=blockSize) {
154                     kBound = Math.min(kk + blockSize, cols1);
155 
156                     // Multiply current blocks.
157                     for(int i=ii; i<iBound; i++) {
158                         src1Start = i*cols1 + kk;
159                         stopIndex = src1Start+(kBound-kk);
160                         destStart = i*cols2;
161 
162                         for (int j=jj; j<jBound; j++) {
163                             destIndex = destStart + j;
164                             src1Index = src1Start;
165                             src2Index = kk*cols2 + j;
166 
167                             while(src1Index < stopIndex) {
168                                 dest[destIndex].addEq(src1[src1Index++].mult(src2[src2Index]));
169                                 src2Index+=cols2;
170                             }
171                         }
172                     }
173                 }
174             }
175         }
176 
177         return dest;
178     }
