34 /**
35  * This class contains methods for computing real or complex Householder reflectors (also known as elementary reflectors).
36  * A Householder reflector is a transformation matrix which reflects a vector about a hyperplane containing the origin.
37  */
38 public class Householder {
39 
40     private Householder() {
41         // Hide default constructor for utility class.
42         throw new IllegalStateException(ErrorMessages.getUtilityClassErrMsg());
43     }

46     /**
47      * Computes the Householder reflector which describes a reflection through a hyperplane containing the origin which
48      * is normal to the specified {@code normal} vector.
49      * @param normal The vector normal to the plane the Householder reflector will reflect through\.
50      * @return A transformation matrix which describes a reflection through a plane containing the origin with the
51      * specified {@code normal} vector, i.e. a Householder reflector.
52      */
53     public static Matrix getReflector(Vector normal) {
54         Matrix H = Matrix.I(normal.size);
55         Vector v = normal.copy();
56 
57         double signedNorm = -Math.copySign(v.norm(), v.entries[0]);
58         v = v.div(v.entries[0] - signedNorm);
59         v.entries[0] = 1;
60 
61         // Create projection matrix
62         Matrix P = v.outer(v).mult(2.0/v.inner(v));
63         H.subEq(P);
64 
65         return H;
66     }

69     /**
70      * Computes the Householder reflector which describes a reflection through a hyperplane containing the origin which
71      * is normal to the specified {@code normal} vector.
72      * @param normal The vector normal to the plane the Householder reflector will reflect through\.
73      * @return A transformation matrix which describes a reflection through a plane containing the origin with the
74      * specified {@code normal} vector, i.e. a Householder reflector.
75      */
76     public static CMatrix getReflector(CVector normal) {
77         CMatrix H = CMatrix.I(normal.size);
78         CVector v = normal.copy();
79 
80         // Compute signed norm using modified sgn function.
81         CNumber signedNorm = v.entries[0].equals(CNumber.ZERO) ?
82                 new CNumber(-v.norm()) : CNumber.sgn(v.entries[0]).mult(-v.norm());
83 
84         v = v.div(v.entries[0].sub(signedNorm));
85         v.entries[0] = new CNumber(1);
86 
87         // Create projection matrix
88         CMatrix P = v.outer(v).mult(2.0/v.inner(v).re);
89         H.subEq(P);
90 
91         return H;
92     }
