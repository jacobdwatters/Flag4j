324      * Compute the L<sub>p, q</sub> norm of a matrix.
325      * @param src Entries of the matrix.
326      * @param shape Shape of the matrix.
327      * @param p First parameter in L<sub>p, q</sub> norm.
328      * @param q Second parameter in L<sub>p, q</sub> norm.
329      * @return The L<sub>p, q</sub> norm of the matrix.
330      * @throws IllegalArgumentException If {@code p} or {@code q} is less than 1.
331      */
332     public static double matrixNormLpq(CNumber[] src, Shape shape, double p, double q) {
333         ParameterChecks.assertGreaterEq(1, p, q);
334 
335         double norm = 0;
336         double colSum;
337         int rows = shape.dims[Axis2D.row()];
338         int cols = shape.dims[Axis2D.col()];
339 
340         for(int j=0; j<cols; j++) {
341             colSum = 0;
342             for(int i=0; i<rows; i++) {
343                 colSum += (Math.pow(src[i*cols + j].magAsDouble(), p));
344             }
345             norm += Math.pow(colSum, q/p);
346         }
347 
348         return Math.pow(norm, 1/q);
349     }
