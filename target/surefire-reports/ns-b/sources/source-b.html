43     /**
44      * Computes the vector inner product for two complex vectors.
45      * @param src1 Entries of the first vector.
46      * @param src2 Entries of the second vector.
47      * @return The inner product of the two vectors.
48      */
49     public static CNumber innerProduct(CNumber[] src1, CNumber[] src2) {
50         ParameterChecks.assertArrayLengthsEq(src1.length, src2.length);
51         CNumber innerProd = new CNumber();
52 
53         for(int i=0; i<src1.length; i++) {
54             innerProd.addEq(src1[i].mult(src2[i].conj()));
55         }
56 
57         return innerProd;
58     }

63      * Computes the vector outer product between two real dense vectors.
64      * @param src1 Entries of first vector.
65      * @param src2 Entries of second vector.
66      * @return The matrix resulting from the vector outer product.
67      */
68     public static CNumber[] outerProduct(CNumber[] src1, CNumber[] src2) {
69         int destIndex;
70         CNumber[] dest = new CNumber[src1.length*src2.length];
71 
72         for(int i=0; i<src1.length; i++) {
73             destIndex = i*src2.length;
74 
75             for(CNumber cNumber : src2) {
76                 dest[destIndex++] = src1[i].mult(cNumber.conj());
77             }
78         }
79 
80         return dest;
81     }
