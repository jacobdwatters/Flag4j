46     /**
47      * Checks if two real dense matrices are equal.
48      * @param A First matrix.
49      * @param B Second matrix.
50      * @return True if the two matrices are element-wise equivalent.
51      */
52     public static boolean matrixEquals(CMatrix A, CMatrix B) {
53         return tensorEquals(A.entries, A.shape, B.entries, B.shape);
54     }

57     /**
58      * Checks if two dense tensors are equal.
59      * @param src1 Entries of first tensor.
60      * @param shape1 Shape of first tensor.
61      * @param src2 Entries of second tensor.
62      * @param shape2 Shape of second tensor.
63      * @return True if the two tensors are numerically element-wise equivalent.
64      */
65     public static boolean tensorEquals(CNumber[] src1, Shape shape1, CNumber[] src2, Shape shape2) {
66         return shape1.equals(shape2) && Arrays.equals(src1, src2);
67     }
