44     /**
45      * Creates a {@code 4x4} perspective projection matrix to transform a 3D point represented in homogeneous
46      * coordinates.
47      *
48      * @param fov Field of view in radians (this is the fov in both the {@code x} and {@code y} directions).
49      *            For distinct field of views see {@link #getPerspective(double, double, double, double, double)}.
50      * @param aspectRatio Aspect ratio of the image plane to project to (i.e. {@code width/height}).
51      * @param nearClip The distance from the camera to the near clipping plane.
52      * @param farClip The distance from the camera to the far clipping plane.
53      * @return The perspective projection matrix based on the provided attributes.
54      * @throws IllegalArgumentException If {@code aspectRatio}  is not positive.
55      * @throws AssertionError If {@code nearClip!=farClip}.
56      */
57     public static Matrix getPerspective(double fov, double aspectRatio, double nearClip, double farClip) {
58         ParameterChecks.assertGreaterEq(0, aspectRatio);
59         assert(nearClip!=farClip) : "nearClip cannot equal farClip.";
60 
61         Matrix perspective = new Matrix(4);
62 
63         double cotHalfFov = 1.0/Math.tan(fov/2.0);
64 
65         perspective.entries[0] = cotHalfFov/aspectRatio;
66         perspective.entries[5] = cotHalfFov;
67         perspective.entries[10] = -(farClip + nearClip)/(farClip - nearClip);
68         perspective.entries[11] = -1;
69         perspective.entries[14] = -2*farClip*nearClip/(farClip - nearClip);
70 
71         return perspective;
72     }

76      * Creates a {@code 4x4} perspective projection matrix to transform a 3D point represented in homogeneous
77      * coordinates.
78      *
79      * @param fovX Field of view, in radians, in the {@code x} direction.
80      * @param fovY Field of view, in radians, in the {@code y} direction.
81      * @param aspectRatio Aspect ratio of the image plane to project to (i.e. {@code width/height}).
82      * @param nearClip The distance from the camera to the near clipping plane.
83      * @param farClip The distance from the camera to the far clipping plane.
84      * @return The perspective projection matrix based on the provided attributes.
85      * @throws IllegalArgumentException If {@code aspectRatio}  is not positive.
86      * @throws AssertionError If {@code nearClip!=farClip}.
87      */
88     public static Matrix getPerspective(double fovX, double fovY, double aspectRatio, double nearClip, double farClip){
89         ParameterChecks.assertGreaterEq(0, aspectRatio);
90         assert(nearClip!=farClip) : "nearClip cannot equal farClip.";
91 
92         Matrix perspective = new Matrix(4);
93 
94         // Convert the field of views to radians.
95         double fovXRad = Math.toRadians(fovX);
96         double fovYRad = Math.toRadians(fovX);
97 
98         perspective.entries[0] = 1.0/(aspectRatio*Math.tan(fovXRad/2.0));
99         perspective.entries[5] = 1.0/(Math.tan(fovYRad/2.0));
100         perspective.entries[10] = -(farClip + nearClip)/(farClip - nearClip);
101         perspective.entries[11] = -1;
102         perspective.entries[14] = -2*farClip*nearClip/(farClip-nearClip);
103 
104         return perspective;
105     }

108     /**
109      * Creates a {@code 4x4} orthogonal projection matrix to project a 3D point in homogeneous coordinates
110      * onto the specified 2D coordinate grid (i.e. image plane).
111      * This is an orthographic projection meaning the distance from the virtual camera will not affect the projection.
112      * @param xMin Minimum {@code x} value of image plane to project to.
113      * @param xMax Maximum {@code x} value of image plane to project to.
114      * @param yMin Minimum {@code y} value of image plane to project to.
115      * @param yMax Maximum {@code y} value of image plane to project to.
116      * @param nearClip Distance from camera to near clipping plane.
117      * @param farClip Distance from camera to far clipping plane.
118      * @return The orthogonal projection for the specified parameters.
119      */
120     public static Matrix getOrthogonal(double xMin, double xMax, double yMin, double yMax,
121                                        double nearClip, double farClip){
122         Matrix ortho = new Matrix(4);
123 
124         ortho.entries[0] = 2.0/(xMax-xMin);
125         ortho.entries[5] = 2.0/(yMax-yMin);
126         ortho.entries[10] = -2.0/(farClip-nearClip);
127         ortho.entries[12] = -(xMax + xMin)/(xMax - xMin);
128         ortho.entries[13] = -(yMax + yMin)/(yMax - yMin);
129         ortho.entries[14] = -(farClip + nearClip)/(farClip-nearClip);
130         ortho.entries[15] = 1;
131 
132         return ortho;
133     }
