39     /**

44      * The default block size for blocked algorithms.
45      */

48      * The default minimum recursive size for recursive algorithms.
49      */

63     /**
64      * Sets the number of threads for use in concurrent operations as the number of processors available to the Java
65      * virtual machine. Note that this value may change during runtime. This method will include logical cores so the value
66      * returned may be higher than the number of physical cores on the machine if hyper-threading is enabled.
67      * <br><br>
68      * This is implemented as: <code>numThreads = {@link Runtime#availableProcessors() Runtime.getRuntime().availableProcessors()};</code>
69      * @return The new value of numThreads, i.e. the number of available processors.
70      */
71     static int setNumThreadsAsAvailableProcessors() {
72         ThreadManager.threadPool = new ForkJoinPool(Runtime.getRuntime().availableProcessors());
73         return ThreadManager.threadPool.getParallelism();
74     }
