34 /**
35  * This class contains the base thread pool for all concurrent operations and several methods for managing the
36  * pool.
37  */
38 public class ThreadManager {

40     /**
41      * Thread pool for managing threads executing concurrent operations.
42      */
43     protected static ForkJoinPool threadPool = new ForkJoinPool();

46     /**
47      * Applies a concurrent loop to a function.
48      * @param startIndex Starting index for concurrent loop.
49      * @param endIndex Ending index for concurrent loop (exclusive).
50      * @param function Function to apply each iteration. Function may be dependent on iteration index but should
51      *                 individual iterations should be independent of each other.
52      */
53     public static void concurrentLoop(int startIndex, int endIndex, IntConsumer function) {
54         try {
55             threadPool.submit(() -> IntStream.range(startIndex, endIndex).parallel().forEach(function)).get();
56         } catch (InterruptedException | ExecutionException e) {
57             e.printStackTrace();
58             Thread.currentThread().interrupt();
59         }
60     }

64     /**
65      * Applies a concurrent loop to a function.
66      * @param startIndex Starting index for concurrent loop.
67      * @param endIndex Ending index for concurrent loop (exclusive).
68      * @param step Step size for the index variable of the loop.
69      * @param function Function to apply each iteration. Function may be dependent on iteration index but should
70      *      individual iterations should be independent of each other.
71      */
72     public static void concurrentLoop(int startIndex, int endIndex, int step, IntConsumer function ) {
73         if (step <= 0)
74             throw new IllegalArgumentException(ErrorMessages.negValueErr(startIndex));
75         try {
76             int range = endIndex - startIndex;
77             int iterations = range/step + ((range%step == 0) ? 0 : 1);
78             threadPool.submit(() -> IntStream.range(0, iterations).parallel().forEach(
79                     i -> function.accept(startIndex + i*step))
80             ).get();
81         } catch (InterruptedException | ExecutionException e) {
82             e.printStackTrace();
83             Thread.currentThread().interrupt();
84         }
85     }
